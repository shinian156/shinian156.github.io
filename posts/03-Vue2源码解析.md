## 二、new Vue() —— 实例初始化流程

```js
// src/core/instance/index.js
function Vue(options) {
  this._init(options)
}

// 混入实例方法（new Vue 时调用）
Vue.prototype._init = function (options) {
  const vm = this
  vm._uid = uid++

  // 1. 标记为 Vue 实例
  vm._isVue = true

  // 2. 合并选项（mixin 支持的核心）
  if (options && options._isComponent) {
    initInternalComponent(vm, options)
  } else {
    vm.$options = mergeOptions(
      resolveConstructorOptions(vm.constructor),
      options || {},
      vm
    )
  }

  // 3. 初始化内部组件
  vm._self = vm
  initLifecycle(vm)     // $parent/$root/$children/$refs
  initEvents(vm)         // $on/$off（父组件传入的事件）
  initRender(vm)         // $createElement/_renderProxy
  callHook(vm, 'beforeCreate')
  initInjections(vm)
  initState(vm)          // 核心：props/data/computed/watch/methods
  initProvide(vm)
  callHook(vm, 'created')

  // 4. 挂载（$mount）
  if (vm.$options.el) {
    vm.$mount(vm.$options.el)
  }
}
```

**执行顺序：** `beforeCreate` → `initState` → `created` → `$mount`

## 三、响应式原理：initState

```js
// src/core/instance/state.js
export function initState(vm) {
  vm._watchers = []
  const opts = vm.$options

  // 初始化顺序：props → methods → data → computed → watch
  if (opts.props) initProps(vm, opts.props)
  if (opts.methods) initMethods(vm, opts.methods)
  if (opts.data) initData(vm, opts.data)
  if (opts.computed) initComputed(vm, opts.computed)
  if (opts.watch) initWatch(vm, opts.watch)
}
```

### initData：数据响应式入口

```js
function initData(vm) {
  let data = vm.$options.data
  data = vm._data = typeof data === 'function' ? data() : data || {}

  // proxy vm.xxx → vm._data.xxx
  const keys = Object.keys(data)
  let i = keys.length
  while (i--) {
    proxy(vm, `_data`, keys[i])
  }

  // observe 递归遍历 data，将所有属性变为响应式
  observe(data, true /* asRootData */)
}
```

## 四、响应式系统：Observer / Dep / Watcher

### Observer：给每个属性添加 getter/setter

```js
// src/core/observer/index.js
export class Observer {
  constructor(value) {
    this.value = value
    this.dep = new Dep()       // 数组依赖（收集$set/$delete触发）
    this.vmCount = 0

    // 将 __ob__ 绑定到 value（可被 Vue.set 访问）
    def(value, '__ob__', this)

    if (Array.isArray(value)) {
      // 数组：改写 7 个会修改数组的方法
      protoAugment(value, arrayMethods)
      this.observeArray(value)
    } else {
      // 对象：walk 遍历所有属性
      this.walk(value)
    }
  }

  walk(obj) {
    const keys = Object.keys(obj)
    for (let i = 0; i < keys.length; i++) {
      defineReactive(obj, keys[i], obj[keys[i]])
    }
  }
}
```

### defineReactive：数据劫持核心

```js
// src/core/observer/index.js
export function defineReactive(obj, key, val) {
  const dep = new Dep()  // 每个属性一个 Dep

  // 递归：如果 val 仍是对象，继续 observe（深度响应式）
  let childOb = observe(val)

  Object.defineProperty(obj, key, {
    enumerable: true,
    configurable: true,
    get() {
      // 读取时，收集依赖
      if (Dep.target) {
        dep.depend()           // watcher 记录这个 Dep
        if (childOb) {
          childOb.dep.depend() // 数组/嵌套对象也收集
        }
      }
      return val
    },
    set(newVal) {
      if (val === newVal) return
      // 设置前先 observe 新值（深度响应式）
      childOb = observe(newVal)
      val = newVal
      // 通知所有依赖的 watcher
      dep.notify()
    }
  })
}
```

### Dep：依赖管理（发布-订阅）

```js
// src/core/observer/dep.js
export class Dep {
  static target = null  // 当前正在计算的 Watcher

  constructor() {
    this.id = uid++
    this.subs = []       // 订阅的 Watcher 数组
  }

  addSub(watcher) {
    this.subs.push(watcher)
  }

  notify() {
    // 依次触发 watcher.update()
    this.subs.forEach(watcher => watcher.update())
  }
}
```

### Watcher：渲染 watcher 与计算 watcher

```js
// src/core/observer/watcher.js
let uid = 0

export class Watcher {
  constructor(vm, expOrFn, cb, options) {
    this.vm = vm
    this.id = ++uid
    this.deep = !!options.deep
    this.user = !!options.user       // 用户 watch
    this.lazy = !!options.lazy        // 计算属性用
    this.sync = !!options.sync        // 同步 watcher
    this.cb = cb
    this.expression = expOrFn.toString()

    if (typeof expOrFn === 'function') {
      this.getter = expOrFn           // 渲染 watcher：执行 updateComponent
    } else {
      this.getter = parsePath(expOrFn) // 用户 watch：解析路径 'a.b.c'
    }

    this.value = this.lazy ? undefined : this.get()
  }

  get() {
    pushTarget(this)  // Dep.target = this
    const value = this.getter.call(this.vm, this.vm)  // 触发 data.xx 的 getter
    popTarget()
    this.traverse(value)
    return value
  }

  update() {
    // 触发时机：dep.notify() → watcher.update()
    if (this.lazy) {
      this.dirty = true          // 计算属性：标记为脏
    } else if (this.sync) {
      this.run()                  // 同步 watcher：直接运行
    } else {
      queueWatcher(this)          // 异步：批量更新（nextTick）
    }
  }

  run() {
    const value = this.get()
    if (value !== this.value) {
      const oldValue = this.value
      this.value = value
      if (this.user) {
        this.cb.call(this.vm, value, oldValue)  // 用户 watch 回调
      }
    }
  }
}
```

## 五、异步更新队列：nextTick

Vue 不是每次数据变更立即更新 DOM，而是批量收集后在下一次微任务中执行：

```js
// src/core/util/next-tick.js
const callbacks = []
let pending = false

function flushCallbacks() {
  pending = false
  const copies = callbacks.slice(0)
  callbacks.length = 0
  copies.forEach(cb => cb())
}

let timerFunc

// 优先级：Promise > MutationObserver > setImmediate > setTimeout
if (typeof Promise !== 'undefined') {
  timerFunc = () => Promise.resolve().then(flushCallbacks)
} else if (typeof MutationObserver !== 'undefined') {
  // ...
} else {
  timerFunc = () => setTimeout(flushCallbacks, 0)
}

export function nextTick(cb, ctx) {
  let _resolve
  callbacks.push(() => {
    if (cb) cb.call(ctx)
    else if (_resolve) _resolve(ctx)
  })

  if (!pending) {
    pending = true
    timerFunc()
  }

  return new Promise(resolve => { _resolve = resolve })
}
```

**在 Watcher 中的使用：**

```js
// src/core/observer/scheduler.js
export function queueWatcher(watcher) {
  const id = watcher.id
  if (has[id] == null) {
    has[id] = true
    queue.push(watcher)

    if (!waiting) {
      waiting = true
      nextTick(flushSchedulerQueue)
    }
  }
}

function flushSchedulerQueue() {
  // 按 id 排序（父组件先于子组件更新）
  queue.sort((a, b) => a.id - b.id)
  for (let i = 0; i < queue.length; i++) {
    watcher = queue[i]
    watcher.run()
  }
}
```

## 六、虚拟 DOM 与 diff 算法

### patch VNode → 真实 DOM

```js
// src/core/vdom/patch.js
function createElm(vnode) {
  const { vnode.tag, data, children, text } = vnode

  if (typeof vtag === 'string') {
    vnode.elm = document.createElement(vtag)
    createChildren(vnode, children)
    updateAttrs(vnode)
    updateClass(vnode)
    updateStyle(vnode)
    updateDOMListeners(vnode)
    // ...
  } else if (isDef(vnode.componentOptions)) {
    // 组件 VNode
    createComponent(vnode)
  }
  return vnode.elm
}
```

### 核心 diff（sameVnode + patchVnode）

```js
function patchVnode(oldVnode, vnode) {
  if (oldVnode === vnode) return

  // 更新文本节点
  if (vnode.text) {
    if (oldVnode.text !== vnode.text) {
      nodeOps.setTextContent(elm, vnode.text)
    }
    return
  }

  // 更新子节点数组
  const oldCh = oldVnode.children
  const ch = vnode.children

  if (oldCh && ch) {
    // 双方都有子节点 → patchChildren（diff）
    if (oldCh !== ch) {
      updateChildren(elm, oldCh, ch)
    }
  } else if (ch) {
    // 新增子节点
    createChildren(vnode, ch)
  } else if (oldCh) {
    // 删除旧子节点
    removeVnodes(oldCh, 0, oldCh.length)
  }
}
```

### updateChildren：同层比较策略

Vue2 diff 采用**两端比对 + Map 辅助**的算法：

```js
function updateChildren(parentElm, oldCh, newCh) {
  let oldStartIdx = 0
  let oldEndIdx = oldCh.length - 1
  let newStartIdx = 0
  let newEndIdx = newCh.length - 1

  let oldStartVnode = oldCh[0]
  let oldEndVnode = oldCh[oldEndIdx]
  let newStartVnode = newCh[0]
  let newEndVnode = newCh[newEndIdx]

  while (oldStartIdx <= oldEndIdx && newStartIdx <= newEndIdx) {
    if (sameVnode(oldStartVnode, newStartVnode)) {
      // 相同节点 → 递归 patch
      patchVnode(oldStartVnode, newStartVnode)
      oldStartVnode = oldCh[++oldStartIdx]
      newStartVnode = newCh[++newStartIdx]

    } else if (sameVnode(oldEndVnode, newEndVnode)) {
      patchVnode(oldEndVnode, newEndVnode)
      oldEndVnode = oldCh[--oldEndIdx]
      newEndVnode = newCh[--newEndIdx]

    } else if (sameVnode(oldStartVnode, newEndVnode)) {
      // 旧头 → 新尾：移动到末尾
      patchVnode(oldStartVnode, newEndVnode)
      parentElm.insertBefore(oldStartVnode.elm, oldEndVnode.elm.nextSibling)
      oldStartVnode = oldCh[++oldStartIdx]
      newEndVnode = newCh[--newEndIdx]

    } else if (sameVnode(oldEndVnode, newStartVnode)) {
      // 旧尾 → 新头：移动到开头
      patchVnode(oldEndVnode, newStartVnode)
      parentElm.insertBefore(oldEndVnode.elm, oldStartVnode.elm)
      oldEndVnode = oldCh[--oldEndIdx]
      newStartVnode = newCh[++newStartIdx]

    } else {
      // 用 Map 匹配 oldCh 中的相同 key
      const keyToOldKeyIdx = createKeyToOldIdx(oldCh)
      const idxInOld = keyToOldKeyIdx[newStartVnode.key]
      if (idxInOld === undefined) {
        // 新节点 → 创建
        createElm(newStartVnode, insertedVnodeQueue)
      } else {
        // 可复用节点 → 移动
        patchVnode(oldCh[idxInOld], newStartVnode)
        oldCh[idxInOld] = undefined
        parentElm.insertBefore(oldCh[idxInOld].elm, oldStartVnode.elm)
      }
      newStartVnode = newCh[++newStartIdx]
    }
  }
}
```

## 七、模板编译：AST → render 函数

### compileToFunctions

```js
// src/platforms/web/compiler/index.js
export function createCompiler(createOptions) {
  return function compile(template, options) {
    const ast = parse(template)              // 模板 → AST
    optimize(ast)                            // 标记静态节点（跳过 patch）
    const code = generate(ast)                // AST → render 字符串
    return {
      ast,
      render: new Function(code),             // new Function → render 函数
      staticRenderFns: code.staticFns
    }
  }
}
```

### generate：AST → with(this){ return _c(...)} 形式

```js
function generate(ast) {
  const state = new CodegenState()
  const code = ast ? genElement(ast, state) : '_c("div")'
  return {
    render: `with(this){return ${code}}`,
    staticRenderFns: state.staticRenderFns
  }
}

function genElement(el, state) {
  if (el.for && !el.ifProcessed) {
    return genFor(el, state)   // v-for → _l(arr, (item,index)=>...)
  } else if (el.if && !el.ifProcessed) {
    return genIf(el, state)    // v-if →三元表达式
  } else if (el.slotTarget) {
    // ...
  } else {
    // 普通元素
    const children = genChildren(el, state)
    return `_c("${el.tag}",${el.attrs ? genProps(el.attrs) : 'undefined'},${
      children ? genChildren(el, state) : 'undefined'
    })`
  }
}
```

## 八、Computed 计算属性原理

```js
// src/core/instance/state.js
function createComputedGetter(key) {
  return function computedGetter() {
    const watcher = this._computedWatchers && this._computedWatchers[key]
    if (watcher) {
      if (watcher.dirty) {
        watcher.evaluate()   // 求值，并标记 dirty=false
      }
      if (Dep.target) {
        watcher.depend()     // 让渲染 watcher 也订阅这些 dep
      }
      return watcher.value
    }
  }
}
```

**watcher.evaluate() = this.get()**：执行 getter，触发所有 data 属性的 getter，从而收集依赖。

## 九、Watch 用户 Watcher 原理

```js
function createWatcher(vm, expOrFn, handler, options) {
  if (isPlainObject(handler)) {
    options = handler
    handler = handler.handler
  }
  if (typeof handler === 'string') {
    handler = vm[handler]  // methods 中的方法名
  }
  return vm.$watch(expOrFn, handler, options)
}

Vue.prototype.$watch = function(expOrFn, cb, options) {
  const vm = this
  const watcher = new Watcher(vm, expOrFn, cb, {
    deep: !!options.deep,
    user: true,            // user watcher
    sync: !!options.sync,
    before: options.before
  })

  if (options.immediate) {
    cb.call(vm, watcher.value)  // 立即执行一次
  }

  return function unwatchFn() {
    watcher.teardown()
  }
}
```

**teardown**：从所有 Dep 中移除该 watcher，实现**取消监听**。
