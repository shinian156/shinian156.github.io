## 一、栈（Stack）

### 概念与实现

栈是一种**后进先出（LIFO）**的数据结构，只允许在一端（栈顶）进行插入和删除。

```js
class Stack {
  constructor() {
    this.items = []
  }

  push(item) { this.items.push(item) }     // 入栈
  pop() { return this.items.pop() }        // 出栈
  peek() { return this.items[this.items.length - 1] }  // 查看栈顶
  isEmpty() { return this.items.length === 0 }
  size() { return this.items.length }
}
```

## 二、栈的经典应用

### 1. 有效的括号

```js
function isValid(s) {
  const stack = []
  const map = { ')': '(', ']': '[', '}': '{' }

  for (const ch of s) {
    if ('([{'.includes(ch)) {
      stack.push(ch)
    } else {
      if (stack.pop() !== map[ch]) return false
    }
  }
  return stack.length === 0
}

isValid('({[]})')   // true
isValid('([)]')     // false
```

### 2. 单调栈（找最近更大/更小元素）

```js
// 每日温度（下一个更大的距离）
function dailyTemperatures(t) {
  const n = t.length
  const ans = new Array(n).fill(0)
  const stack = []  // 存索引，单调递增温度

  for (let i = 0; i < n; i++) {
    while (stack.length && t[i] > t[stack[stack.length - 1]]) {
      const prev = stack.pop()
      ans[prev] = i - prev
    }
    stack.push(i)
  }
  return ans
}

// 柱状图中最大矩形
function largestRectangleArea(heights) {
  const stack = []
  let max = 0

  for (let i = 0; i <= heights.length; i++) {
    while (stack.length && heights[i] < heights[stack[stack.length - 1]]) {
      const h = heights[stack.pop()]
      const w = stack.length === 0 ? i : i - stack[stack.length - 1] - 1
      max = Math.max(max, h * w)
    }
    stack.push(i)
  }
  return max
}
```

### 3. 表达式求值（中缀 → 后缀）

```js
// 后缀表达式求值
function evalRPN(tokens) {
  const stack = []
  const ops = { '+': (a,b)=>a+b, '-':(a,b)=>a-b, '*':(a,b)=>a*b, '/':(a,b)=>Math.trunc(a/b) }

  for (const t of tokens) {
    if (ops[t]) {
      const b = stack.pop(), a = stack.pop()
      stack.push(ops[t](a, b))
    } else {
      stack.push(Number(t))
    }
  }
  return stack[0]
}

evalRPN(["2","1","+","3","*"])  // (2+1)*3 = 9
```

## 三、队列（Queue）

### 基础实现

```js
class Queue {
  constructor() {
    this.items = []
  }
  enqueue(item) { this.items.push(item) }
  dequeue() { return this.items.shift() }
  front() { return this.items[0] }
  isEmpty() { return this.items.length === 0 }
  size() { return this.items.length }
}

// 性能优化：用循环队列
class CircularQueue {
  constructor(k) {
    this.k = k
    this.queue = new Array(k)
    this.head = -1
    this.tail = -1
  }

  enqueue(val) {
    if (this.isFull()) return false
    this.tail = (this.tail + 1) % this.k
    this.queue[this.tail] = val
    if (this.head === -1) this.head = this.tail
    return true
  }

  dequeue() {
    if (this.isEmpty()) return false
    if (this.head === this.tail) {
      this.head = -1; this.tail = -1
    } else {
      this.head = (this.head + 1) % this.k
    }
    return true
  }
}
```

## 五、滑动窗口最大值

```js
// 方案1：单调递减队列（最优）
function maxSlidingWindow(nums, k) {
  const deque = []  // 存索引，保持单调递减
  const result = []

  for (let i = 0; i < nums.length; i++) {
    // 移除窗口外的
    if (deque.length && deque[0] <= i - k) deque.shift()

    // 移除比当前小的（不可能成为最大值）
    while (deque.length && nums[deque[deque.length - 1]] < nums[i]) {
      deque.pop()
    }

    deque.push(i)

    if (i >= k - 1) result.push(nums[deque[0]])
  }
  return result
}
```

## 六、优先队列（Priority Queue）

```js
class PriorityQueue {
  constructor(comparator = (a, b) => a > b) {
    this.heap = []
    this.comparator = comparator
  }

  size() { return this.heap.length }

  push(val) {
    this.heap.push(val)
    this.bubbleUp(this.heap.length - 1)
  }

  pop() {
    if (this.heap.length === 0) return null
    const top = this.heap[0]
    const last = this.heap.pop()
    if (this.heap.length > 0) {
      this.heap[0] = last
      this.bubbleDown(0)
    }
    return top
  }

  bubbleUp(i) {
    while (i > 0) {
      const parent = Math.floor((i - 1) / 2)
      if (this.comparator(this.heap[i], this.heap[parent])) {
        [this.heap[i], this.heap[parent]] = [this.heap[parent], this.heap[i]]
        i = parent
      } else break
    }
  }

  bubbleDown(i) {
    const n = this.heap.length
    while (true) {
      let smallest = i
      const left = 2 * i + 1, right = 2 * i + 2
      if (left < n && this.comparator(this.heap[left], this.heap[smallest])) smallest = left
      if (right < n && this.comparator(this.heap[right], this.heap[smallest])) smallest = right
      if (smallest !== i) {
        [this.heap[i], this.heap[smallest]] = [this.heap[smallest], this.heap[i]]
        i = smallest
      } else break
    }
  }
}

// 最小堆：取最小元素
const minPQ = new PriorityQueue((a, b) => a < b)

// 最大堆：取最大元素
const maxPQ = new PriorityQueue((a, b) => a > b)

// Top K 问题
function topKFrequent(nums, k) {
  const freq = {}
  nums.forEach(n => freq[n] = (freq[n] || 0) + 1)

  const pq = new PriorityQueue((a, b) => a[1] < b[1])
  for (const [num, count] of Object.entries(freq)) {
    pq.push([num, count])
    if (pq.size() > k) pq.pop()
  }

  return pq.heap.map(([n]) => Number(n))
}
```

## 七、经典算法应用

### BFS 广度优先搜索（用队列）

```js
function levelOrder(root) {
  if (!root) return []
  const queue = [root]
  const result = []

  while (queue.length) {
    const levelSize = queue.length
    const level = []
    for (let i = 0; i < levelSize; i++) {
      const node = queue.shift()
      level.push(node.val)
      if (node.left) queue.push(node.left)
      if (node.right) queue.push(node.right)
    }
    result.push(level)
  }
  return result
}
```

## 八、栈与队列对比

| 特性 | 栈 | 队列 | 双端队列 |
|------|---|---|---|
| 插入 | push O(1) | enqueue O(1) | 两端 O(1) |
| 删除 | pop O(1) | dequeue O(1) | 两端 O(1) |
| 访问 | peek O(1) | front O(1) | 两端 O(1) |
| 核心思想 | LIFO | FIFO | 两端均可 |
| 主要用途 | 递归、括号、单调性 | BFS、任务调度 | 滑动窗口 |
