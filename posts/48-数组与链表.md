## 一、数组（Array）

### 基本概念

数组是一种**连续的固定大小**的线性存储结构，通过索引（从 0 开始）可直接访问任意元素。

```
内存布局：
索引:    0     1     2     3     4
值:    [12]  [23]  [8]   [56]  [34]
地址:  0x100 0x104 0x108 0x10C 0x110

特点：
✅ 随机访问 O(1)：arr[i] 直接通过 base + i * size 计算地址
✅ CPU 缓存友好：连续内存，访问局部性好
❌ 插入/删除 O(n)：需要移动后续元素
❌ 大小固定（普通数组）：扩容需要重新分配
```

### 常用操作复杂度

| 操作 | 时间复杂度 |
|------|-----------|
| 随机访问 | O(1) |
| 尾部插入 | O(1)（均摊）|
| 尾部删除 | O(1) |
| 中间插入 | O(n) |
| 中间删除 | O(n) |
| 查找元素 | O(n) |

## 二、数组的高级用法

### 二分查找

```js
// 二分查找（有序数组）
function binarySearch(arr, target) {
  let left = 0, right = arr.length - 1

  while (left <= right) {
    const mid = Math.floor((left + right) / 2)

    if (arr[mid] === target) return mid
    if (arr[mid] < target) left = mid + 1
    else right = mid - 1
  }

  return -1  // 未找到
}

binarySearch([1, 3, 5, 7, 9, 11], 7)  // → 3
// 时间：O(logn)  空间：O(1)
```

### 双指针

```js
// 有序数组去重
function removeDuplicates(nums) {
  if (nums.length === 0) return 0

  let slow = 0
  for (let fast = 1; fast < nums.length; fast++) {
    if (nums[fast] !== nums[slow]) {
      slow++
      nums[slow] = nums[fast]
    }
  }
  return slow + 1
}

// 两数之和（有序数组）
function twoSum(arr, target) {
  let left = 0, right = arr.length - 1
  while (left < right) {
    const sum = arr[left] + arr[right]
    if (sum === target) return [left, right]
    if (sum < target) left++
    else right--
  }
  return []
}

// 滑动窗口最大值
function maxSlidingWindow(nums, k) {
  const deque = []   // 存索引
  const result = []

  for (let i = 0; i < nums.length; i++) {
    // 移除窗口外的元素
    if (deque.length > 0 && deque[0] <= i - k) deque.shift()

    // 保持单调递减
    while (deque.length > 0 && nums[deque[deque.length - 1]] < nums[i]) {
      deque.pop()
    }
    deque.push(i)

    // 窗口满了记录结果
    if (i >= k - 1) result.push(nums[deque[0]])
  }
  return result
}
```

### 合并区间

```js
function merge(intervals) {
  if (intervals.length <= 1) return intervals

  // 按起点排序
  intervals.sort((a, b) => a[0] - b[0])

  const result = [intervals[0]]

  for (let i = 1; i < intervals.length; i++) {
    const last = result[result.length - 1]
    const cur = intervals[i]

    if (cur[0] <= last[1]) {
      // 有重叠，合并
      last[1] = Math.max(last[1], cur[1])
    } else {
      result.push(cur)
    }
  }
  return result
}

merge([[1,3],[2,6],[8,10],[15,18]])
// → [[1,6],[8,10],[15,18]]
```

## 三、链表（Linked List）

### 单向链表

```js
class ListNode {
  constructor(val, next = null) {
    this.val = val
    this.next = next
  }
}

class LinkedList {
  constructor() {
    this.head = null
    this.size = 0
  }

  // 头部插入 O(1)
  prepend(val) {
    const node = new ListNode(val, this.head)
    this.head = node
    this.size++
  }

  // 尾部插入 O(n)
  append(val) {
    const node = new ListNode(val)
    if (!this.head) { this.head = node; this.size++; return }
    let curr = this.head
    while (curr.next) curr = curr.next
    curr.next = node
    this.size++
  }

  // 按索引删除 O(n)
  removeAt(index) {
    if (index < 0 || index >= this.size) return null
    if (index === 0) {
      const removed = this.head
      this.head = this.head.next
      this.size--
      return removed
    }
    let curr = this.head
    for (let i = 0; i < index - 1; i++) curr = curr.next
    const removed = curr.next
    curr.next = curr.next.next
    this.size--
    return removed
  }
}
```

### 快慢指针技巧

```js
// 1. 找链表中点
function middleNode(head) {
  let slow = head, fast = head
  while (fast && fast.next) {
    slow = slow.next
    fast = fast.next.next
  }
  return slow  // 中点（偶数个返回下中点）
}

// 2. 链表是否有环（环形链表）
function hasCycle(head) {
  let slow = head, fast = head
  while (fast && fast.next) {
    slow = slow.next
    fast = fast.next.next
    if (slow === fast) return true
  }
  return false
}

// 3. 找环入口
function detectCycle(head) {
  let slow = head, fast = head
  while (fast && fast.next) {
    slow = slow.next
    fast = fast.next.next
    if (slow === fast) {
      // 找入口：从头和相遇点同时出发，相遇处即入口
      let start = head
      while (start !== slow) {
        start = start.next
        slow = slow.next
      }
      return start
    }
  }
  return null
}

// 4. 找倒数第 k 个节点
function kthFromEnd(head, k) {
  let slow = head, fast = head
  for (let i = 0; i < k; i++) fast = fast.next
  while (fast) { slow = slow.next; fast = fast.next }
  return slow
}

// 5. 合并两个有序链表
function mergeTwoLists(l1, l2) {
  const dummy = new ListNode(0)
  let curr = dummy
  while (l1 && l2) {
    if (l1.val <= l2.val) { curr.next = l1; l1 = l1.next }
    else { curr.next = l2; l2 = l2.next }
    curr = curr.next
  }
  curr.next = l1 || l2
  return dummy.next
}
```

### 反转链表

```js
// 迭代反转（空间 O(1)）
function reverseList(head) {
  let prev = null, curr = head
  while (curr) {
    const next = curr.next
    curr.next = prev
    prev = curr
    curr = next
  }
  return prev
}

// 递归反转（空间 O(n)）
function reverseListRecursive(head) {
  if (!head || !head.next) return head
  const newHead = reverseListRecursive(head.next)
  head.next.next = head
  head.next = null
  return newHead
}

// 部分反转（m 到 n）
function reverseBetween(head, m, n) {
  const dummy = new ListNode(0, head)
  let prev = dummy
  for (let i = 0; i < m - 1; i++) prev = prev.next

  let curr = prev.next
  for (let i = 0; i < n - m; i++) {
    const next = curr.next
    curr.next = next.next
    next.next = prev.next
    prev.next = next
  }
  return dummy.next
}
```

### 双向链表（LRU Cache 实现）

```js
class LRUCache {
  constructor(capacity) {
    this.capacity = capacity
    this.map = new Map()
    this.head = new Node(0, 0)
    this.tail = new Node(0, 0)
    this.head.next = this.tail
    this.tail.prev = this.head
  }

  get(key) {
    if (!this.map.has(key)) return -1
    const node = this.map.get(key)
    this.remove(node)
    this.addToHead(node)
    return node.val
  }

  put(key, val) {
    if (this.map.has(key)) {
      this.remove(this.map.get(key))
    }
    const node = new Node(key, val)
    this.addToHead(node)
    this.map.set(key, node)
    if (this.map.size > this.capacity) {
      const removed = this.tail.prev
      this.remove(removed)
      this.map.delete(removed.key)
    }
  }

  remove(node) {
    node.prev.next = node.next
    node.next.prev = node.prev
  }

  addToHead(node) {
    node.next = this.head.next
    node.prev = this.head
    this.head.next.prev = node
    this.head.next = node
  }
}
```

## 四、数组 vs 链表对比

| 特性 | 数组 | 链表 |
|------|------|------|
| 内存 | 连续 | 不连续（分散）|
| 随机访问 | O(1) | O(n) |
| 头部插入 | O(n) | O(1) |
| 尾部插入 | O(1)（均摊）| O(n) |
| 中间插入 | O(n) | O(1)（已知位置）|
| 内存利用率 | 高（无额外开销）| 低（需存指针）|
| CPU 缓存 | 友好 | 不友好 |
| 实现难度 | 简单 | 较复杂 |
