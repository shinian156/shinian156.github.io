## 1. Koa2 基础概念

### 1.1 什么是 Koa2？
Koa2 是一个新的 web 框架，由 Express 团队打造，致力于成为一个更小、更富有表现力、更健壮的 Web 框架。Koa2 使用 ES6+ 的 async/await 语法，使异步代码更易读。

### 1.2 核心特点
- **轻量级**：核心库只有约 600 行代码
- **异步支持**：使用 async/await 处理异步
- **洋葱模型**：中间件执行顺序清晰
- **上下文（Context）**：统一的请求/响应对象
- **模块化**：功能通过中间件扩展

### 1.3 Koa vs Express
| 特性 | Koa | Express |
|------|-----|---------|
| 异步处理 | async/await | 回调 |
| 中间件模式 | 洋葱模型 | 线性模型 |
| 内置功能 | 极少 | 丰富 |
| 路由 | 需要插件 | 内置 |
| 轻量级 | 是 | 否 |
| 学习曲线 | 较陡 | 平缓 |

## 2. Koa2 基础使用

### 2.1 快速开始
```javascript
const Koa = require('koa');
const app = new Koa();

// 中间件
app.use(async ctx => {
  ctx.body = 'Hello Koa2';
});

app.listen(3000);
```

### 2.2 中间件使用
```javascript
const Koa = require('koa');
const app = new Koa();

// 中间件1：日志
app.use(async (ctx, next) => {
  console.log(`请求方法: ${ctx.method}`);
  console.log(`请求路径: ${ctx.url}`);
  await next();
  console.log(`响应状态: ${ctx.status}`);
});

// 中间件2：响应时间
app.use(async (ctx, next) => {
  const start = Date.now();
  await next();
  const ms = Date.now() - start;
  ctx.set('X-Response-Time', `${ms}ms`);
});

// 中间件3：业务逻辑
app.use(async ctx => {
  ctx.body = 'Hello Koa2';
});

app.listen(3000);
```

### 2.3 路由处理
```javascript
const Koa = require('koa');
const Router = require('@koa/router');
const app = new Koa();
const router = new Router();

// 定义路由
router.get('/', async ctx => {
  ctx.body = 'Home Page';
});

router.get('/users', async ctx => {
  ctx.body = 'Users List';
});

router.get('/users/:id', async ctx => {
  ctx.body = `User ${ctx.params.id}`;
});

router.post('/users', async ctx => {
  ctx.body = 'Create User';
});

// 使用路由
app.use(router.routes());
app.use(router.allowedMethods());

app.listen(3000);
```

## 3. 中间件详解

### 3.1 洋葱模型
```javascript
const Koa = require('koa');
const app = new Koa();

// 中间件1
app.use(async (ctx, next) => {
  console.log('1 -> 进入中间件1');
  await next();
  console.log('1 -> 离开中间件1');
});

// 中间件2
app.use(async (ctx, next) => {
  console.log('2 -> 进入中间件2');
  await next();
  console.log('2 -> 离开中间件2');
});

// 中间件3
app.use(async ctx => {
  console.log('3 -> 处理请求');
  ctx.body = 'Hello Koa2';
  console.log('3 -> 响应完成');
});

app.listen(3000);

// 输出顺序：
// 1 -> 进入中间件1
// 2 -> 进入中间件2
// 3 -> 处理请求
// 3 -> 响应完成
// 2 -> 离开中间件2
// 1 -> 离开中间件1
```

### 3.2 常用中间件

#### Koa-router（路由）
```javascript
const Router = require('@koa/router');
const router = new Router();

router.get('/', async ctx => {
  ctx.body = 'Hello';
});

app.use(router.routes());
```

#### Koa-bodyparser（解析请求体）
```javascript
const bodyParser = require('koa-bodyparser');
app.use(bodyParser());

// 访问请求体
app.use(async ctx => {
  const body = ctx.request.body;
  ctx.body = body;
});
```

#### Koa-static（静态资源）
```javascript
const serve = require('koa-static');
app.use(serve(__dirname + '/public'));
```

#### Koa-json（JSON 格式化）
```javascript
const json = require('koa-json');
app.use(json());
```

#### Koa-logger（日志）
```javascript
const logger = require('koa-logger');
app.use(logger());
```

### 3.3 自定义中间件
```javascript
// 错误处理中间件
app.use(async (ctx, next) => {
  try {
    await next();
  } catch (err) {
    ctx.status = err.status || 500;
    ctx.body = {
      error: err.message
    };
    ctx.app.emit('error', err, ctx);
  }
});

// 认证中间件
async function auth(ctx, next) {
  const token = ctx.header.authorization;
  
  if (!token) {
    ctx.throw(401, 'Token is required');
  }
  
  // 验证 token
  const user = await verifyToken(token);
  ctx.state.user = user;
  
  await next();
}

app.use(auth);
```

## 4. 请求和响应

### 4.1 请求处理
```javascript
app.use(async ctx => {
  // 获取查询参数
  const query = ctx.query;
  const name = ctx.query.name;
  
  // 获取路由参数
  const params = ctx.params;
  const id = ctx.params.id;
  
  // 获取请求头
  const headers = ctx.header;
  const userAgent = ctx.get('user-agent');
  
  // 获取请求体
  const body = ctx.request.body;
  
  // 获取请求方法
  const method = ctx.method;
  
  // 获取请求路径
  const path = ctx.path;
  
  // 获取 IP
  const ip = ctx.ip;
});
```

### 4.2 响应处理
```javascript
app.use(async ctx => {
  // 设置响应体
  ctx.body = {
    message: 'Hello Koa2'
  };
  
  // 设置状态码
  ctx.status = 200;
  
  // 设置响应头
  ctx.set('Content-Type', 'application/json');
  ctx.set('X-Custom-Header', 'value');
  
  // 重定向
  ctx.redirect('/login');
  
  // 下载文件
  ctx.attachment('file.pdf');
});
```

## 5. Context 对象

### 5.1 常用属性
```javascript
app.use(async ctx => {
  // 请求相关
  ctx.request;      // 请求对象
  ctx.method;       // 请求方法
  ctx.url;          // 请求 URL
  ctx.query;        // 查询参数
  ctx.querystring;  // 查询字符串
  ctx.header;       // 请求头
  
  // 响应相关
  ctx.response;     // 响应对象
  ctx.body;         // 响应体
  ctx.status;       // 状态码
  ctx.type;         // Content-Type
  ctx.length;       // Content-Length
  
  // 状态存储
  ctx.state;        // 中间件间传递数据
  ctx.app;          // 应用实例
  
  // 其他
  ctx.ip;           // 客户端 IP
  ctx.cookies;      // Cookie 操作
  ctx.throw();      // 抛出错误
});
```

### 5.2 Cookie 操作
```javascript
app.use(async ctx => {
  // 设置 Cookie
  ctx.cookies.set('name', 'value', {
    httpOnly: true,
    maxAge: 3600000,
    secure: true,
    sameSite: 'strict'
  });
  
  // 获取 Cookie
  const name = ctx.cookies.get('name');
  
  // 删除 Cookie
  ctx.cookies.set('name', null, { maxAge: -1 });
});
```

## 6. 错误处理

### 6.1 错误处理中间件
```javascript
app.use(async (ctx, next) => {
  try {
    await next();
  } catch (err) {
    ctx.status = err.status || 500;
    ctx.body = {
      code: err.status,
      message: err.message,
      data: null
    };
    
    // 开发环境打印错误堆栈
    if (process.env.NODE_ENV === 'development') {
      ctx.body.stack = err.stack;
    }
    
    // 触发应用错误事件
    ctx.app.emit('error', err, ctx);
  }
});

// 监听错误事件
app.on('error', (err, ctx) => {
  console.error('Server error:', err);
});
```

### 6.2 自定义错误
```javascript
class AppError extends Error {
  constructor(message, status = 500) {
    super(message);
    this.status = status;
    this.name = 'AppError';
  }
}

// 使用
app.use(async ctx => {
  if (!ctx.query.name) {
    throw new AppError('Name is required', 400);
  }
});
```

## 7. 数据库集成

```javascript
const mongoose = require('mongoose');

// 连接数据库
mongoose.connect('mongodb://localhost:27017/myapp', {
  useNewUrlParser: true,
  useUnifiedTopology: true
});

const User = mongoose.model('User', {

  email: String

  // 创建

  // 查询
  const users = await User.find();
  
  ctx.body = users;
});
```

## 8. 文件上传

### 8.1 使用 koa-multer
```javascript
const multer = require('@koa/multer');

// 配置上传
const upload = multer({
  dest: 'uploads/',
  limits: {
    fileSize: 5 * 1024 * 1024 // 5MB
  }
});

// 单文件上传
app.post('/upload', upload.single('file'), async ctx => {
  ctx.body = {
    filename: ctx.file.filename,
    originalname: ctx.file.originalname
  };
});

// 多文件上传
app.post('/upload', upload.array('files', 10), async ctx => {
  ctx.body = ctx.files;
});
```

### 8.2 使用 koa-body
```javascript
const koaBody = require('koa-body');

app.use(koaBody({
  multipart: true,
  formidable: {
    uploadDir: __dirname + '/uploads',
    keepExtensions: true
  }
}));

app.post('/upload', async ctx => {
  const files = ctx.request.files;
  ctx.body = files;
});
```

## 9. 会话管理

### 9.1 使用 koa-session
```javascript
const session = require('koa-session');

app.keys = ['some secret'];

app.use(session({
  key: 'koa.sess',
  maxAge: 86400000,
  overwrite: true,
  httpOnly: true,
  signed: true
}, app));

app.use(async ctx => {
  // 设置会话
  ctx.session.user = { id: 1, name: 'Alice' };
  
  // 获取会话
  const user = ctx.session.user;
  
  ctx.body = user;
});
```

### 9.2 JWT 认证
```javascript
const jwt = require('jsonwebtoken');

// 生成 Token
const token = jwt.sign(
  { id: user.id, name: user.name },
  'secret',
  { expiresIn: '1h' }
);

// 验证 Token
function verifyToken(token) {
  return jwt.verify(token, 'secret');
}

// 认证中间件
async function auth(ctx, next) {
  const token = ctx.header.authorization;
  
  if (!token) {
    ctx.throw(401, 'Token is required');
  }
  
  try {
    const decoded = verifyToken(token);
    ctx.state.user = decoded;
    await next();
  } catch (err) {
    ctx.throw(401, 'Invalid token');
  }
}

app.use(auth);
```

## 10. RESTful API 设计

### 10.1 标准结构
```javascript
const Router = require('@koa/router');
const router = new Router({ prefix: '/api/users' });

// 获取用户列表
router.get('/', async ctx => {
  const users = await User.find();
  ctx.body = users;
});

// 获取单个用户
router.get('/:id', async ctx => {
  const user = await User.findById(ctx.params.id);
  if (!user) {
    ctx.throw(404, 'User not found');
  }
  ctx.body = user;
});

// 创建用户
router.post('/', async ctx => {
  const user = await User.create(ctx.request.body);
  ctx.body = user;
  ctx.status = 201;
});

// 更新用户
router.put('/:id', async ctx => {
  const user = await User.findByIdAndUpdate(
    ctx.params.id,
    ctx.request.body,
    { new: true }
  );
  ctx.body = user;
});

// 删除用户
router.delete('/:id', async ctx => {
  await User.findByIdAndDelete(ctx.params.id);
  ctx.status = 204;
});

app.use(router.routes());
```

## 11. 面试高频题

### Q1: Koa 的洋葱模型是什么？
- 中间件执行顺序：先进后出
- 类似洋葱，层层包裹
- 可以在请求前和响应后执行代码

### Q2: Koa 和 Express 的区别？
- Koa 使用 async/await，Express 使用回调
- Koa 中间件是洋葱模型，Express 是线性模型
- Koa 更轻量，Express 功能更丰富
- Koa 学习曲线较陡

### Q3: 如何实现错误处理？
- try-catch 捕获错误
- ctx.throw() 抛出错误
- app.on('error') 监听错误

### Q4: 如何处理跨域？
```javascript
const cors = require('@koa/cors');
app.use(cors());
```

### Q5: 如何实现请求日志？
```javascript
app.use(async (ctx, next) => {
  const start = Date.now();
  await next();
  const ms = Date.now() - start;
  console.log(`${ctx.method} ${ctx.url} - ${ms}ms`);
});
```

### Q6: 如何实现限流？

const rateLimit = require('koa-ratelimit');
