## 二、基础排序（O(n²)）

### 冒泡排序

```js
// 每轮把最大冒到末尾
function bubbleSort(arr) {
  const n = arr.length
  for (let i = 0; i < n - 1; i++) {
    let swapped = false
    for (let j = 0; j < n - i - 1; j++) {
      if (arr[j] > arr[j + 1]) {
        [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]]
        swapped = true
      }
    }
    if (!swapped) break  // 已有序，提前终止
  }
  return arr
}
// 最优 O(n)：数组已有序时，一趟比较发现无交换，直接退出
```

### 选择排序

```js
// 每轮选最小，放到已排序区末尾
function selectionSort(arr) {
  const n = arr.length
  for (let i = 0; i < n - 1; i++) {
    let minIdx = i
    for (let j = i + 1; j < n; j++) {
      if (arr[j] < arr[minIdx]) minIdx = j
    }
    [arr[i], arr[minIdx]] = [arr[minIdx], arr[i]]
  }
  return arr
}
// 不稳定：e.g. [2,2,1] → [1,2,2]，两个2的相对位置改变
```

### 插入排序

```js
// 像打扑克牌，逐张插入已排序区
function insertionSort(arr) {
  for (let i = 1; i < arr.length; i++) {
    const cur = arr[i]
    let j = i - 1
    while (j >= 0 && arr[j] > cur) {
      arr[j + 1] = arr[j]
      j--
    }
    arr[j + 1] = cur
  }
  return arr
}
// 最优 O(n)：数组已有序，每次内层循环立即终止
```

## 三、分治排序

### 归并排序

```js
function mergeSort(arr) {
  if (arr.length <= 1) return arr

  const mid = Math.floor(arr.length / 2)
  const left = mergeSort(arr.slice(0, mid))
  const right = mergeSort(arr.slice(mid))

  return merge(left, right)
}

function merge(left, right) {
  const result = []
  let i = 0, j = 0
  while (i < left.length && j < right.length) {
    if (left[i] <= right[j]) result.push(left[i++])
    else result.push(right[j++])
  }
  return result.concat(left.slice(i)).concat(right.slice(j))
}
// 空间 O(n)，稳定！适合链表（不需要随机访问）
```

### 快速排序（最重要！）

```js
function quickSort(arr) {
  if (arr.length <= 1) return arr

  const pivot = arr[Math.floor(arr.length / 2)]
  const left = arr.filter(x => x < pivot)
  const middle = arr.filter(x => x === pivot)
  const right = arr.filter(x => x > pivot)

  return [...quickSort(left), ...middle, ...quickSort(right)]
}

// 原地版本（空间 O(logn)）
function quickSortInPlace(arr, low = 0, high = arr.length - 1) {
  if (low < high) {
    const pi = partition(arr, low, high)
    quickSortInPlace(arr, low, pi - 1)
    quickSortInPlace(arr, pi + 1, high)
  }
  return arr
}

function partition(arr, low, high) {
  const pivot = arr[high]
  let i = low - 1
  for (let j = low; j < high; j++) {
    if (arr[j] < pivot) {
      i++
      [arr[i], arr[j]] = [arr[j], arr[i]]
    }
  }
  [arr[i + 1], arr[high]] = [arr[high], arr[i + 1]]
  return i + 1
}

// 最优 O(nlogn)，最坏 O(n²)（已排序 + 取首/尾作 pivot）
// 优化：随机选 pivot 或 三数取中
```

## 四、堆排序

```js
function heapSort(arr) {
  const n = arr.length

  // 构建最大堆（从最后一个非叶子节点向上）
  for (let i = Math.floor(n / 2) - 1; i >= 0; i--) {
    heapify(arr, n, i)
  }

  // 逐个提取最大到末尾
  for (let i = n - 1; i > 0; i--) {
    [arr[0], arr[i]] = [arr[i], arr[0]]  // 交换堆顶和末尾
    heapify(arr, i, 0)  // 重新堆化
  }
  return arr
}

function heapify(arr, n, i) {
  let largest = i
  const left = 2 * i + 1
  const right = 2 * i + 2

  if (left < n && arr[left] > arr[largest]) largest = left
  if (right < n && arr[right] > arr[largest]) largest = right
  if (largest !== i) {
    [arr[i], arr[largest]] = [arr[largest], arr[i]]
    heapify(arr, n, largest)
  }
}
// 时间始终 O(nlogn)，空间 O(1)，不稳定
```

## 五、特殊排序（线性时间）

### 计数排序

```js
// 适用于范围不大的整数排序
function countingSort(arr) {
  if (!arr.length) return arr

  const min = Math.min(...arr)
  const max = Math.max(...arr)
  const range = max - min + 1
  const count = new Array(range).fill(0)
  const output = new Array(arr.length)

  // 计数
  for (let i = 0; i < arr.length; i++) count[arr[i] - min]++

  // 前缀和（保持稳定性）
  for (let i = 1; i < range; i++) count[i] += count[i - 1]

  // 倒序填入（稳定性关键）
  for (let i = arr.length - 1; i >= 0; i--) {
    output[--count[arr[i] - min]] = arr[i]
  }

  return output
}
// 时间 O(n+k)，空间 O(k)，稳定
```

### 桶排序

```js
function bucketSort(arr, bucketSize = 5) {
  if (arr.length === 0) return arr

  const min = Math.min(...arr)
  const max = Math.max(...arr)
  const bucketCount = Math.ceil((max - min) / bucketSize)
  const buckets = Array.from({ length: bucketCount }, () => [])

  // 分配到桶
  for (const val of arr) {
    const idx = Math.floor((val - min) / bucketSize)
    buckets[idx].push(val)
  }

  // 每桶内部排序，再合并
  const result = []
  for (const bucket of buckets) {
    bucket.sort((a, b) => a - b)
    result.push(...bucket)
  }
  return result
}
// 适用于均匀分布的浮点数排序
```

### 基数排序

```js
function radixSort(arr) {
  const max = Math.max(...arr)

  // 从低位到高位，按位计数排序
  for (let exp = 1; Math.floor(max / exp) > 0; exp *= 10) {
    countingSortByDigit(arr, exp)
  }
  return arr
}

function countingSortByDigit(arr, exp) {
  const n = arr.length
  const output = new Array(n)
  const count = new Array(10).fill(0)

  for (const val of arr) {
    const digit = Math.floor(val / exp) % 10
    count[digit]++
  }
  for (let i = 1; i < 10; i++) count[i] += count[i - 1]

  for (let i = n - 1; i >= 0; i--) {
    const digit = Math.floor(arr[i] / exp) % 10
    output[--count[digit]] = arr[i]
  }
  for (let i = 0; i < n; i++) arr[i] = output[i]
}
// 按位处理，每位用计数排序，时间 O(nk)（k=位数）
```

## 六、实战选择指南

```js
// 根据数据特征选排序算法：

// 1. 数据量小（< 50）：插入排序
//    优势：O(n²) 系数小，小数据量下反而最快；且对近乎有序的数据近乎 O(n)

// 2. 数据量中等（50 ~ 10k）：快速排序 / 归并排序
//    快速排序：原地排序，空间 O(logn)，实际最快
//    归并排序：稳定，适合外部排序（大数据量文件排序）

// 3. 数据量大：外部排序 / 归并排序 + 计数排序
//    内存放不下时，分块读入排序再合并

// 4. 整数且范围不大：计数排序 → O(n)

// 5. 浮点数均匀分布：桶排序

// 6. 通用场景：JavaScript 引擎（V8）使用 TimSort（插入排序 + 归并排序）
//    Python sort() 也用 TimSort
```
