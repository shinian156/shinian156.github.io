## 一、量化交易概述

量化交易（Quantitative Trading）是利用数学模型和计算机程序来制定、执行交易决策的交易方式，通过数据分析、统计建模和算法执行来替代主观判断。

### 量化交易 vs 主观交易

| 维度 | 量化交易 | 主观交易 |
|---|---|---|
| 决策依据 | 数学模型/历史数据 | 经验/直觉 |
| 情绪影响 | 无 | 有 |
| 执行速度 | 毫秒级 | 人工执行 |
| 策略数量 | 可同时运行多个 | 精力有限 |
| 可回测 | 是 | 难以量化 |
| 适应性 | 需要迭代调整 | 灵活 |

### 量化交易的核心价值

1. **纪律性**：严格按规则执行，不受情绪干扰
2. **可复现**：策略逻辑可以精确描述和重复
3. **可扩展**：同一策略可以同时跑多个标的
4. **可回测**：在历史数据上验证策略有效性

## 二、量化交易核心概念

### 交易标的

- **股票**：A 股（沪深市场）、港股、美股
- **期货**：商品期货、股指期货、国债期货
- **期权**：认购/认沽期权
- **数字货币**：BTC、ETH 等
- **外汇**：货币对交易

### 重要术语

| 术语 | 含义 |
|---|---|
| Alpha | 超额收益，相对基准的超出部分 |
| Beta | 市场敏感度，衡量与市场的相关性 |
| 夏普比率 | 超额收益 / 波动率，越高越好 |
| 最大回撤 | 历史最高点到最低点的最大跌幅 |
| 回测 | 用历史数据检验策略表现 |
| 过拟合 | 策略在历史数据表现好但实盘差 |
| 滑点 | 理论成交价与实际成交价的差距 |
| 换手率 | 持仓变动频率，高换手对应高交易成本 |

### 数据预处理

```python
import pandas as pd
import numpy as np

def preprocess_data(df):
    """数据清洗与预处理"""
    df = df.copy()
    df['日期'] = pd.to_datetime(df['日期'])
    df = df.sort_values('日期').reset_index(drop=True)

    # 处理缺失值
    df = df.dropna(subset=['收盘', '成交量'])
    df['成交量'] = df['成交量'].fillna(0)

    # 计算收益率
    df['return'] = df['收盘'].pct_change()
    df['log_return'] = np.log(df['收盘'] / df['收盘'].shift(1))

    # 删除首行 NaN
    df = df.dropna().reset_index(drop=True)
    return df
```

## 五、技术指标计算

### 常用指标实现

```python
import pandas as pd
import numpy as np

def calculate_indicators(df):
    close = df['收盘']
    high = df['最高']
    low = df['最低']
    volume = df['成交量']

    # 移动平均线
    df['MA5'] = close.rolling(5).mean()
    df['MA10'] = close.rolling(10).mean()
    df['MA20'] = close.rolling(20).mean()
    df['MA60'] = close.rolling(60).mean()

    # EMA 指数移动平均
    df['EMA12'] = close.ewm(span=12, adjust=False).mean()
    df['EMA26'] = close.ewm(span=26, adjust=False).mean()

    # MACD
    df['MACD'] = df['EMA12'] - df['EMA26']
    df['Signal'] = df['MACD'].ewm(span=9, adjust=False).mean()
    df['Histogram'] = df['MACD'] - df['Signal']

    # RSI 相对强弱指标
    delta = close.diff()
    gain = delta.clip(lower=0).rolling(14).mean()
    loss = (-delta.clip(upper=0)).rolling(14).mean()
    rs = gain / loss
    df['RSI14'] = 100 - 100 / (1 + rs)

    # 布林带
    df['BB_Middle'] = close.rolling(20).mean()
    std = close.rolling(20).std()
    df['BB_Upper'] = df['BB_Middle'] + 2 * std
    df['BB_Lower'] = df['BB_Middle'] - 2 * std

    # KDJ
    low_n = low.rolling(9).min()
    high_n = high.rolling(9).max()
    rsv = (close - low_n) / (high_n - low_n + 1e-9) * 100
    df['K'] = rsv.ewm(alpha=1/3, adjust=False).mean()
    df['D'] = df['K'].ewm(alpha=1/3, adjust=False).mean()
    df['J'] = 3 * df['K'] - 2 * df['D']

    # ATR 真实波动幅度
    tr = pd.concat([
        high - low,
        (high - close.shift()).abs(),
        (low - close.shift()).abs()
    ], axis=1).max(axis=1)
    df['ATR14'] = tr.rolling(14).mean()

    # 成交量均线
    df['VOL_MA5'] = volume.rolling(5).mean()
    df['VOL_MA20'] = volume.rolling(20).mean()

    return df
```

## 六、回测框架基础

### 手写简单回测引擎

```python
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

class SimpleBacktest:
    def __init__(self, df, initial_capital=100000):
        self.df = df.copy()
        self.initial_capital = initial_capital
        self.capital = initial_capital
        self.position = 0        # 持仓数量
        self.trades = []         # 交易记录
        self.portfolio_values = []

    def buy(self, date, price, quantity=None):
        """买入"""
        if quantity is None:
            quantity = int(self.capital * 0.95 / price / 100) * 100  # 95% 资金，整手
        cost = quantity * price * 1.0003  # 手续费 0.03%
        if self.capital >= cost and quantity > 0:
            self.capital -= cost
            self.position += quantity
            self.trades.append({
                'date': date, 'type': 'BUY',
                'price': price, 'quantity': quantity
            })

    def sell(self, date, price):
        """卖出全部"""
        if self.position > 0:
            revenue = self.position * price * (1 - 0.0013)  # 手续费+印花税
            self.capital += revenue
            self.trades.append({
                'date': date, 'type': 'SELL',
                'price': price, 'quantity': self.position
            })
            self.position = 0

    def run(self, strategy_func):
        """运行策略"""
        for i, row in self.df.iterrows():
            signal = strategy_func(self.df.iloc[:i+1], self.position)
            if signal == 'BUY' and self.position == 0:
                self.buy(row['日期'], row['收盘'])
            elif signal == 'SELL' and self.position > 0:
                self.sell(row['日期'], row['收盘'])
            # 记录组合价值
            portfolio_val = self.capital + self.position * row['收盘']
            self.portfolio_values.append(portfolio_val)

    def get_stats(self):
        """计算绩效指标"""
        values = np.array(self.portfolio_values)
        total_return = (values[-1] - self.initial_capital) / self.initial_capital
        n_days = len(values)
        annual_return = (1 + total_return) ** (252 / n_days) - 1
        daily_returns = np.diff(values) / values[:-1]
        volatility = daily_returns.std() * np.sqrt(252)
        sharpe = (annual_return - 0.02) / volatility if volatility > 0 else 0

        # 最大回撤
        running_max = np.maximum.accumulate(values)
        drawdown = (running_max - values) / running_max
        max_dd = drawdown.max()

        return {
            '总收益率': f'{total_return:.2%}',
            '年化收益率': f'{annual_return:.2%}',
            '年化波动率': f'{volatility:.2%}',
            '夏普比率': f'{sharpe:.2f}',
            '最大回撤': f'{max_dd:.2%}',
            '交易次数': len(self.trades)
        }


# 使用示例：双均线策略
def ma_crossover_strategy(df, position):
    if len(df) < 20:
        return 'HOLD'
    ma5 = df['收盘'].rolling(5).mean().iloc[-1]
    ma20 = df['收盘'].rolling(20).mean().iloc[-1]
    ma5_prev = df['收盘'].rolling(5).mean().iloc[-2]
    ma20_prev = df['收盘'].rolling(20).mean().iloc[-2]

    if ma5 > ma20 and ma5_prev <= ma20_prev:  # 金叉
        return 'BUY'
    elif ma5 < ma20 and ma5_prev >= ma20_prev:  # 死叉
        return 'SELL'
    return 'HOLD'
```

def factor_scoring(stock_df):
    """多因子综合评分"""
    scores = pd.DataFrame()
    # PE 因子（越低越好，取倒数排名）
    scores['pe_score'] = stock_df['PE'].rank(ascending=True)
    # 动量因子（过去3月涨幅）
    scores['mom_score'] = stock_df['return_3m'].rank(ascending=False)
    # 综合评分
    scores['total'] = scores['pe_score'] + scores['mom_score']
    return scores.sort_values('total', ascending=False)

## 八、风险管理

量化交易中风险管理比信号生成更重要：

```python
class RiskManager:
    def __init__(self, max_drawdown=0.2, position_limit=0.1):
        self.max_drawdown = max_drawdown      # 最大允许回撤
        self.position_limit = position_limit  # 单股最大仓位

    def calc_position_size(self, capital, price, volatility, risk_per_trade=0.02):
        """基于波动率的仓位计算（固定风险法）"""
        # 每笔交易最多亏损 capital * risk_per_trade
        stop_loss = price * volatility * 2   # 2倍 ATR 止损
        if stop_loss <= 0:
            return 0
        max_loss = capital * risk_per_trade
        quantity = int(max_loss / stop_loss / 100) * 100
        # 不超过最大仓位限制
        max_quantity = int(capital * self.position_limit / price / 100) * 100
        return min(quantity, max_quantity)

    def should_stop(self, current_value, peak_value):
        """是否触发最大回撤止损"""
        drawdown = (peak_value - current_value) / peak_value
        return drawdown >= self.max_drawdown
```

### 风险管理原则

1. **严格止损**：每笔交易设置止损位，亏损不超过 1~2%
2. **分散投资**：单只股票持仓不超过总资金 10%
3. **控制回撤**：整体回撤超过 20% 时停止交易
4. **避免过拟合**：用样本外数据验证策略
5. **小资金验证**：新策略先用小资金实盘测试
