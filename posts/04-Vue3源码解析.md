## 二、Vue3 响应式原理：Proxy 替代 Object.defineProperty

### reactive

```js
// packages/reactivity/src/reactive.ts
export function reactive(target) {
  // 如果是只读的，直接返回
  if (target && (target as Target).__v_isReadonly) {
    return target
  }
  return createReactiveObject(
    target,
    false,           // isReadonly
    mutableHandlers, // Proxy handlers
    reactiveMap
  )
}

function createReactiveObject(
  target: Target,
  isReadonly: boolean,
  baseHandlers: ProxyHandler<any>,
  proxyMap: WeakMap<Target, any>
) {
  // 已代理过的对象，直接返回缓存
  const existingProxy = proxyMap.get(target)
  if (existingProxy) return existingProxy

  const proxy = new Proxy(target, baseHandlers)
  proxyMap.set(target, proxy)
  return proxy
}
```

### Proxy Handlers（get / set / deleteProperty）

```js
// packages/reactivity/src/baseHandlers.ts
export const mutableHandlers: ProxyHandler<Record<string, any>> = {
  get(target, key, receiver) {
    const res = Reflect.get(target, key, receiver)

    // 收集依赖
    if (isRef(res)) {
      track(target, TrackOpTypes.GET, key)
      return res.value  // ref 自动解包
    }

    track(target, TrackOpTypes.GET, key)

    // 递归深度响应式
    return isObject(res) ? reactive(res) : res
  },

  set(target, key, value, receiver) {
    const oldValue = target[key]
    const result = Reflect.set(target, key, value, receiver)

    if (!hadKey) {
      // 新增属性
      trigger(target, TriggerOpTypes.ADD, key, value)
    } else if (hasChanged(value, oldValue)) {
      // 修改属性
      trigger(target, TriggerOpTypes.SET, key, value, oldValue)
    }

    return result
  },

  deleteProperty(target, key) {
    const hadKey = hasOwn(target, key)
    const result = Reflect.deleteProperty(target, key)
    if (result && hadKey) {
      trigger(target, TriggerOpTypes.DELETE, key)
    }
    return result
  },

  has(target, key) {
    track(target, TrackOpTypes.HAS, key)
    return Reflect.has(target, key)
  },

  ownKeys(target) {
    track(target, TrackOpTypes.ITERATE, isArray(target) ? 'length' : ITERATE_KEY)
    return Reflect.ownKeys(target)
  }
}
```

**相比 Vue2 的优势：**
- 不需要 `Object.defineProperty`，原生支持数组下标
- 不需要 `Vue.set`，新增属性自动响应
- 不需要深度遍历初始化，一次 Proxy 拦截全部

## 三、依赖追踪：track / trigger

```js
// packages/reactivity/src/dep.ts

// 全局当前活跃的 effect（相当于 Vue2 的 Dep.target）
export let activeEffect: ReactiveEffect | undefined

export const targetMap = new WeakMap<Object, Map<string | symbol, Set<ReactiveEffect>>>()

// 读取属性时调用：在 targetMap 中记录"哪个对象+属性"对应"哪些 effect"
export function track(target: object, operationType: TrackOpTypes, key?: string | symbol) {
  if (!activeEffect) return  // 没有 active effect 则跳过

  let depsMap = targetMap.get(target)
  if (!depsMap) {
    targetMap.set(target, (depsMap = new Map()))
  }

  let dep = depsMap.get(key!)
  if (!dep) {
    depsMap.set(key!, (dep = new Set()))
  }

  dep.add(activeEffect)    // effect 订阅这个 dep
  activeEffect.deps.push(dep)  // 反向记录：dep 在 effect 中（用于清理）
}

// 写入属性时调用：找到对应的 dep，触发所有 effect
export function trigger(
  target: object,
  type: TriggerOpTypes,
  key?: string | symbol,
  newValue?: any
) {
  const depsMap = targetMap.get(target)
  if (!depsMap) return

  const dep = depsMap.get(key!)
  if (dep) {
    dep.forEach(effect => effect.run())
  }

  if (type === TriggerOpTypes.ADD && isArray(target) && key === 'length') {
    // 数组 length 缩短时，触发所有索引 >= length 的 effect
    depsMap.forEach((dep, key) => {
      if (key !== 'length' && key !== Symbol('isShallow')) {
        dep.forEach(effect => effect.run())
      }
    })
  }
}
```

## 四、effect 与 ReactiveEffect

```js
// packages/reactivity/src/effect.ts
export function effect<T = any>(
  fn: () => T,
  options?: ReactiveEffectOptions
) {
  const _effect = new ReactiveEffect(fn)
  if (options) {
    Object.assign(_effect, options)
  }

  if (!options || !options.lazy) {
    _effect.run()  // 立即执行一次，建立依赖
  }

  return _effect
}

export class ReactiveEffect<T = any> {
  active = true
  deps: Dep[] = []
  onStop?: () => void

  constructor(
    public fn: () => T,
    public scheduler?: () => void
  ) {}

  run() {
    if (!this.active) {
      return this.fn()  // 已 stop 时，直接执行
    }

    activeEffect = this  // 设置当前活跃 effect
    try {
      return this.fn()   // 执行，触发 Proxy get，track 收集依赖
    } finally {
      activeEffect = undefined
    }
  }

  stop() {
    if (this.active) {
      cleanupEffect(this)  // 从所有 dep 中移除
      this.onStop?.()
      this.active = false
    }
  }
}
```

## 五、ref 与 computed

### ref

```js
// packages/reactivity/src/ref.ts
export function ref<T>(value: T) {
  return createRef(value, false)
}

function createRef(rawValue: T, shallow: boolean) {
  if (isRef(rawValue)) return rawValue  // 已经是 ref 则直接返回

  return new RefImpl(rawValue)
}

class RefImpl<T> {
  public dep?: Dep = undefined
  private _rawValue: T
  private _value: T

  constructor(value: T) {
    this._rawValue = value
    this._value = toReactive(value)  // 对象类型自动 reactive
  }

  get value() {
    trackRefValue(this)  // 收集依赖
    return this._value
  }

  set value(newVal) {
    if (hasChanged(newVal, this._rawValue)) {
      this._rawValue = newVal
      this._value = toReactive(newVal)
      triggerRefValue(this, TriggerOpTypes.SET, newVal)  // 触发更新
    }
  }
}
```

### computed

```js
// packages/reactivity/src/computed.ts
export function computed<T>(getter: ComputedGetter<T>) {
  return new ComputedRefImpl(getter)
}

class ComputedRefImpl<T> {
  public dep?: Dep = undefined
  private _value!: T
  private _dirty = true  // 初始为脏

  constructor(getter: ComputedGetter<T>) {
    // computed 内部使用 effect，不立即执行
    this.effect = effect(getter, {
      scheduler: () => {
        if (!this._dirty) {
          this._dirty = true
          triggerRefValue(this, TriggerOpTypes.SET, 'value')
        }
      }
    })
  }

  get value() {
    trackRefValue(this)
    if (this._dirty) {
      this._dirty = false
      this._value = this.effect.run()!
    }
    return this._value
  }
}
```

**computed 的惰性求值**：`lazy: true` + `_dirty` 标志，只有读取时才求值，且值不变时不重新计算。

## 六、renderer：组件渲染与更新

```js
// packages/runtime-core/src/renderer.ts

// patch 是 Vue3 的核心渲染函数，处理所有 VNode 类型
const patch: PatchFn = (n1, n2, container, anchor, parentComponent, ...) => {
  const { type, shapeFlags } = n2

  if (type === Text) {
    // 文本节点
  } else if (type === Fragment) {
    // Fragment
  } else if (shapeFlags & ShapeFlags.ELEMENT) {
    processElement(n1, n2, container, anchor, parentComponent, ...)
  } else if (shapeFlags & ShapeFlags.STATEFUL_COMPONENT) {
    processComponent(n1, n2, container, anchor, parentComponent, ...)
  }
}

// 组件更新
const processComponent = (n1, n2, container, anchor, parentComponent) => {
  if (n1 === null) {
    mountComponent(n2, container, anchor, parentComponent)
  } else {
    updateComponent(n1, n2, parentComponent)  // 关键更新路径
  }
}

const updateComponent = (n1, n2, parentComponent) => {
  const instance = n2.component = n1.component!
  // 更新 props（会触发 Proxy setter → trigger）
  instance.updateProps(n1.props, n2.props)

  // 调度更新：异步批量
  instance.update()
}
```

## 七、patchFlag 编译优化

Vue3 模板编译器在编译阶段**分析模板**，对动态内容加上标记，运行时只处理动态部分：

```js
// 模板
<div class="container" style="color:red">
  <h1>{{ title }}</h1>
  <p>{{ count }}</p>
</div>
```

编译后：

```js
import { createElementVNode as _createElementVNode, toDisplayString as _toDisplayString, openBlock as _openBlock, createElementBlock as _createElementBlock } from "vue"

export function render(_ctx, _cache) {
  return (_openBlock(), _createElementBlock("div", {
    class: "container",
    style: {"color":"red"}
  }, [
    _createElementVNode("h1", null, _toDisplayString(_ctx.title), 1 /* TEXT */),
    //       ^^^^^^^^ patchFlag=1：只有文本内容是动态的
    _createElementVNode("p", null, _toDisplayString(_ctx.count), 1 /* TEXT */)
  ], 2 /* CLASS, STYLE */))
  //                                                        ^^^^^^^^^^^^^^^^
  //                                                   patchFlag=2：class 和 style 是动态的
}
```

### patchFlag 类型

| 值 | 含义 |
|---|---|
| 1 | TEXT（动态文本） |
| 2 | CLASS |
| 4 | STYLE |
| 8 | PROPS |
| 16 | FULL_PROPS（props 全量更新） |
| 32 | NEED_PATCH（需要完整 patch） |
| 64 | HYDRATE_EVENTS |
| 128 | STABLE_FRAGMENT |
| 256 | KEYED_FRAGMENT |
| 512 | UNKEYED_FRAGMENT |

## 八、Block 与动态节点收集

Vue3 引入 **Block** 概念：在 `openBlock()` 时创建一个动态节点数组，`createElementVNode` 动态节点时自动推入数组：

```js
// createElementBlock
function createElementBlock(type, props, children, patchFlag, ... ) {
  const vnode = createBaseVNode(type, props, children, patchFlag, ...)
  blockStack.push(vnode)  // 当前 block 入栈
  return vnode
}

// 动态子节点收集
function createElementVNode(...) {
  return _createElementVNode(..., patchFlag, ...)  // 有 patchFlag 则为动态
}

function _createElementVNode(...) {
  const vnode = { ..., patchFlag, dynamicChildren: null }

  // 关键：如果当前在 block 内，添加到动态列表
  if (currentBlock && patchFlag > 0) {
    currentBlock.push(vnode)
  }
  return vnode
}
```

**优势：** 动态节点不再像 Vue2 需要递归 diff，而是**平铺收集到 `dynamicChildren`**，patch 时只需遍历动态节点，数量大幅减少。

## 九、组合式 API（Composition API）实现

```js
// setup(props, { attrs, slots, emit, expose, root }) { ... }

// 生命周期注册（内部实现）
let currentInstance = null

export function setupComponent(instance) {
  const setupResult = setup(instance.props, ...)
  handleSetupResult(instance, setupResult)
}

function handleSetupResult(instance, setupResult) {
  if (typeof setupResult === 'function') {
    instance.render = setupResult  // setup 返回 render 函数
  } else if (isObject(setupResult)) {
    // setup 返回响应式对象 → 挂到 instance.proxy 上
    instance.setupState = setupResult
    // proxy 上通过 getPrototypeByChain 查找 setupState / data / props
  }
}
```

```js
function registerLifecycleHook(instance, type, hook) {

  const invoker = createInvoker(hook)

function invokeArrayFns(fns) {
  for (let i = 0; i < fns.length; i++) {
    fns[i]()  // 调用时从 invoker 取出最新值
  }
}
```

## 十、Vue2 vs Vue3 核心区别

| | Vue2 | Vue3 |
|---|---|---|
| 响应式 | `Object.defineProperty` | `Proxy` |
| 依赖收集 | `Dep.target` 单值 | `activeEffect` + `targetMap` |
| 虚拟 DOM | 全量 diff，递归 | Block + `dynamicChildren` 定向 diff |
| 编译优化 | 无 | `patchFlag` + ` hoistStatic` |
| 动态节点 | 需要递归比对 | 平铺收集 `dynamicChildren` |
| 组件实例 | `this.xxx` | `setup()` 返回对象 → `proxy` |
| 组合式 API | ❌ | ✅ `setup()` + `ref/reactive/computed` |
| TypeScript | 支持但不完善 | 完整重写，类型完善 |
| 包大小 | ~33kb | ~22kb（核心） |
