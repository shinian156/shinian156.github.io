## 二、经典策略一：双均线策略

### 策略逻辑

- 短期均线（MA5）上穿长期均线（MA20）时买入 → **金叉**
- 短期均线（MA5）下穿长期均线（MA20）时卖出 → **死叉**

```python
class MACrossStrategy:
    """双均线策略"""

    def __init__(self, short_window=5, long_window=20, initial_capital=100000):
        self.short_window = short_window
        self.long_window = long_window
        self.capital = initial_capital
        self.initial_capital = initial_capital
        self.position = 0
        self.trades = []

    def generate_signals(self, df: pd.DataFrame) -> pd.DataFrame:
        df = df.copy()
        df['ma_short'] = df['close'].rolling(self.short_window).mean()
        df['ma_long'] = df['close'].rolling(self.long_window).mean()

        # 信号：1 买入, -1 卖出, 0 持有
        df['signal'] = 0
        # 金叉（短线从下方穿越长线）
        df.loc[(df['ma_short'] > df['ma_long']) &
               (df['ma_short'].shift(1) <= df['ma_long'].shift(1)), 'signal'] = 1
        # 死叉（短线从上方穿越长线）
        df.loc[(df['ma_short'] < df['ma_long']) &
               (df['ma_short'].shift(1) >= df['ma_long'].shift(1)), 'signal'] = -1
        return df

    def backtest(self, df: pd.DataFrame) -> dict:
        df = self.generate_signals(df)
        portfolio = []

        for _, row in df.iterrows():
            if row['signal'] == 1 and self.position == 0:
                # 买入：用 90% 资金，整手（100股为一手）
                shares = int(self.capital * 0.9 / row['close'] / 100) * 100
                if shares > 0:
                    cost = shares * row['close'] * 1.0003  # 手续费
                    self.capital -= cost
                    self.position = shares
                    self.trades.append({
                        'date': row['date'], 'action': 'BUY',
                        'price': row['close'], 'shares': shares
                    })

            elif row['signal'] == -1 and self.position > 0:
                # 卖出：含印花税 0.1% + 手续费 0.03%
                revenue = self.position * row['close'] * (1 - 0.0013)
                self.capital += revenue
                self.trades.append({
                    'date': row['date'], 'action': 'SELL',
                    'price': row['close'], 'shares': self.position
                })
                self.position = 0

            total_value = self.capital + self.position * row['close']
            portfolio.append({'date': row['date'], 'value': total_value})

        portfolio_df = pd.DataFrame(portfolio)
        return self._calc_stats(portfolio_df, df)

    def _calc_stats(self, portfolio_df, signal_df):
        values = portfolio_df['value'].values
        total_return = (values[-1] - self.initial_capital) / self.initial_capital
        n_days = len(values)
        annual_return = (1 + total_return) ** (252 / n_days) - 1

        daily_returns = pd.Series(values).pct_change().dropna()
        volatility = daily_returns.std() * np.sqrt(252)
        sharpe = (annual_return - 0.02) / volatility if volatility > 0 else 0

        running_max = np.maximum.accumulate(values)
        max_drawdown = ((running_max - values) / running_max).max()

        # 基准（买入持有）收益
        benchmark = (signal_df['close'].iloc[-1] / signal_df['close'].iloc[0] - 1)

        return {
            'portfolio_df': portfolio_df,
            '总收益率': f'{total_return:.2%}',
            '年化收益率': f'{annual_return:.2%}',
            '基准收益率': f'{benchmark:.2%}',
            '超额收益': f'{total_return - benchmark:.2%}',
            '夏普比率': f'{sharpe:.3f}',
            '最大回撤': f'{max_drawdown:.2%}',
            '交易次数': len(self.trades),
        }


# 使用示例
df = get_stock_data('000001', '20200101', '20231231')
strategy = MACrossStrategy(short_window=5, long_window=20, initial_capital=100000)
result = strategy.backtest(df)
for k, v in result.items():
    if k != 'portfolio_df':
        print(f'{k}: {v}')
```

## 三、经典策略二：RSI 超买超卖策略

### 策略逻辑

- RSI < 30（超卖区）买入，等待反弹
- RSI > 70（超买区）卖出，规避回调

```python
def rsi_strategy(df: pd.DataFrame, period=14, oversold=30, overbought=70) -> pd.DataFrame:
    df = df.copy()

    # 计算 RSI
    delta = df['close'].diff()
    gain = delta.clip(lower=0).rolling(period).mean()
    loss = (-delta.clip(upper=0)).rolling(period).mean()
    rs = gain / (loss + 1e-9)
    df['rsi'] = 100 - 100 / (1 + rs)

    # 信号
    df['signal'] = 0
    df.loc[(df['rsi'] < oversold) & (df['rsi'].shift(1) >= oversold), 'signal'] = 1   # 买入
    df.loc[(df['rsi'] > overbought) & (df['rsi'].shift(1) <= overbought), 'signal'] = -1  # 卖出

    return df


def backtest_rsi(symbol='000001', start='20200101', end='20231231'):
    df = get_stock_data(symbol, start, end)
    df = rsi_strategy(df)

    capital = 100000
    position = 0
    portfolio = []

    for _, row in df.iterrows():
        if pd.isna(row['rsi']):
            portfolio.append(capital)
            continue

        if row['signal'] == 1 and position == 0:
            shares = int(capital * 0.9 / row['close'] / 100) * 100
            if shares > 0:
                capital -= shares * row['close'] * 1.0003
                position = shares

        elif row['signal'] == -1 and position > 0:
            capital += position * row['close'] * 0.9987
            position = 0

        portfolio.append(capital + position * row['close'])

    total_return = (portfolio[-1] - 100000) / 100000
    print(f'RSI策略总收益率: {total_return:.2%}')
    return df, portfolio
```

## 四、经典策略三：布林带策略

### 策略逻辑

- 收盘价突破下轨时买入（价格偏低，期待回归）
- 收盘价突破上轨时卖出（价格偏高，期待回调）

```python
def bollinger_strategy(df: pd.DataFrame, window=20, num_std=2) -> pd.DataFrame:
    df = df.copy()

    df['bb_middle'] = df['close'].rolling(window).mean()
    std = df['close'].rolling(window).std()
    df['bb_upper'] = df['bb_middle'] + num_std * std
    df['bb_lower'] = df['bb_middle'] - num_std * std
    df['bb_width'] = (df['bb_upper'] - df['bb_lower']) / df['bb_middle']  # 带宽

    df['signal'] = 0
    # 价格从下方穿越下轨 → 买入
    df.loc[(df['close'] > df['bb_lower']) &
           (df['close'].shift(1) <= df['bb_lower'].shift(1)), 'signal'] = 1
    # 价格从下方穿越上轨（持仓中） → 卖出
    df.loc[(df['close'] >= df['bb_upper']), 'signal'] = -1

    return df
```

## 五、经典策略四：多因子选股策略

### 策略逻辑

每月末对全市场股票按多因子综合评分排序，买入排名前 N 支，月底换仓。

```python
def multi_factor_select(date: str, top_n=10) -> list:
    """多因子选股：低 PE + 高动量 + 高 ROE"""
    # 获取全市场股票列表
    all_stocks = ak.stock_zh_a_spot_em()

    scores = []
    for _, row in all_stocks.iterrows():
        try:
            code = row['代码']
            pe = row.get('市盈率-动态', None)
            # 获取近3月收益率
            hist = ak.stock_zh_a_hist(symbol=code, period='monthly',
                                       start_date='20230901', end_date=date,
                                       adjust='qfq')
            if len(hist) < 3 or pe is None or pe <= 0:
                continue
            momentum = (hist['收盘'].iloc[-1] / hist['收盘'].iloc[0] - 1)
            scores.append({
                'code': code, 'name': row['名称'],
                'pe': pe, 'momentum': momentum
            })
        except:
            continue

    df = pd.DataFrame(scores)
    if df.empty:
        return []

    # 因子标准化 + 评分
    df['pe_rank'] = df['pe'].rank(ascending=True, pct=True)       # PE 越低越好
    df['mom_rank'] = df['momentum'].rank(ascending=False, pct=True) # 动量越高越好
    df['score'] = df['pe_rank'] * 0.4 + df['mom_rank'] * 0.6      # 加权综合评分

    top_stocks = df.nlargest(top_n, 'score')
    return top_stocks['code'].tolist()
```

## 六、经典策略五：网格交易策略

### 策略逻辑

在一定价格范围内均匀布置买卖单，价格下跌时分批买入，上涨时分批卖出，震荡市表现优秀。

```python
class GridStrategy:
    """网格交易策略"""

    def __init__(self, base_price, grid_pct=0.02, grid_num=10, capital_per_grid=5000):
        """
        base_price: 基准价格
        grid_pct: 每格间距百分比（2%）
        grid_num: 上下各 N 格
        capital_per_grid: 每格使用资金
        """
        self.base_price = base_price
        self.grid_pct = grid_pct
        self.grid_num = grid_num
        self.capital_per_grid = capital_per_grid

        # 生成网格价格列表
        self.buy_grids = [
            base_price * (1 - grid_pct * i) for i in range(1, grid_num + 1)
        ]
        self.sell_grids = [
            base_price * (1 + grid_pct * i) for i in range(1, grid_num + 1)
        ]

        self.holdings = {}   # {价格: 持仓数量}
        self.cash = capital_per_grid * grid_num * 2  # 预留资金
        self.pnl = 0

    def check_signal(self, current_price: float) -> list:
        """检查是否触发网格信号"""
        signals = []

        for buy_price in self.buy_grids:
            if current_price <= buy_price and buy_price not in self.holdings:
                shares = int(self.capital_per_grid / buy_price / 100) * 100
                if shares > 0 and self.cash >= shares * buy_price:
                    signals.append(('BUY', buy_price, shares))
                    self.holdings[buy_price] = shares
                    self.cash -= shares * buy_price

        for sell_price in self.sell_grids:
            # 找到对应的买入格（下方最近的买入价）
            buy_price = sell_price / (1 + self.grid_pct) ** 2
            closest_buy = min(self.holdings.keys(),
                              key=lambda x: abs(x - buy_price),
                              default=None)
            if (current_price >= sell_price and
                    closest_buy and closest_buy in self.holdings):
                shares = self.holdings.pop(closest_buy)
                revenue = shares * sell_price
                self.cash += revenue
                profit = shares * (sell_price - closest_buy)
                self.pnl += profit
                signals.append(('SELL', sell_price, shares, profit))

        return signals

    def get_status(self, current_price):
        position_value = sum(s * current_price for s in self.holdings.values())
        return {
            '现金': self.cash,
            '持仓市值': position_value,
            '总资产': self.cash + position_value,
            '已实现盈亏': self.pnl,
            '持仓网格数': len(self.holdings)
        }
```

```python
from itertools import product

def optimize_ma_strategy(df, short_range=(3, 20, 1), long_range=(10, 60, 5)):
    results = []
```

```python
# 将数据分为训练集（样本内）和测试集（样本外）
def walk_forward_test(df, strategy_class, params, n_splits=5):
    """滚动验证：在多个时间窗口验证策略稳定性"""
    chunk_size = len(df) // n_splits
    results = []

    for i in range(n_splits - 1):
        train = df.iloc[:chunk_size * (i + 1)]
        test = df.iloc[chunk_size * (i + 1): chunk_size * (i + 2)]

        # 在训练集上优化
        strategy = strategy_class(**params)
        train_result = strategy.backtest(train.copy())

        # 在测试集上验证
        strategy_test = strategy_class(**params)
        test_result = strategy_test.backtest(test.copy())

        results.append({
            'period': i + 1,
            'train_return': train_result['总收益率'],
            'test_return': test_result['总收益率']
        })

    return pd.DataFrame(results)
```

```python
            self.short(bar.close_price, self.fixed_size)
```

## 九、量化策略常见陷阱

### 1. 过拟合（Overfitting）

```
症状：回测收益极高，实盘却亏钱
原因：参数在历史数据上过度优化，无法泛化
解决：样本外验证、滚动回测、减少参数数量
```

### 2. 前视偏差（Look-ahead Bias）

```python
# 错误：用了未来数据
df['signal'] = (df['close'] > df['close'].shift(-1)).astype(int)  # 用了明天的收盘价！

# 正确：只用历史数据
df['signal'] = (df['close'] > df['close'].shift(1)).astype(int)
```

### 3. 幸存者偏差（Survivorship Bias）

```
问题：回测时只用了现在还在交易的股票
影响：剔除了历史上退市的亏损股，导致回测收益虚高
解决：使用包含退市股票的完整历史股票池
```

### 4. 交易成本低估

```python
# 实际成本比预想高得多
# A 股买入：手续费 0.02%~0.03%
# A 股卖出：手续费 0.02%~0.03% + 印花税 0.1%（单向）
# 还有：滑点（实际成交价偏差）、冲击成本（大单影响价格）

# 高频策略要格外注意成本对收益的侵蚀
annual_cost = turnover_rate * 0.13%  # 年换手率 * 单次交易成本
```

### 5. 策略容量问题

```
问题：策略在小资金有效，大资金时自身影响价格
解决：测算策略最大可用资金（通常不超过日均成交额的 1/10）
```

### LSTM 预测股价趋势

```python
import torch
import torch.nn as nn

class StockLSTM(nn.Module):
    def __init__(self, input_size, hidden_size=64, num_layers=2, dropout=0.2):
        super().__init__()
        self.lstm = nn.LSTM(input_size, hidden_size, num_layers,
                            batch_first=True, dropout=dropout)
        self.fc = nn.Linear(hidden_size, 1)
        self.sigmoid = nn.Sigmoid()

    def forward(self, x):
        out, _ = self.lstm(x)
        out = self.fc(out[:, -1, :])  # 取最后时间步
        return self.sigmoid(out)
```
