## 一、动态规划核心思想

**动态规划 = 最优子结构 + 重叠子问题 + 状态转移**

1. **大问题拆成小问题**：子问题之间不能相互独立
2. **记住已算过的结果**：避免重复计算（记忆化）
3. **状态转移方程**：dp[i] = f(dp[i-1], dp[i-2], ...)

**判别是否是 DP 问题：**
- 计数：有多少种方式走到终点
- 最值：最大/最小/最长方案
- 存在性：能否达到某个状态

## 二、基础线性 DP

### 爬楼梯（最经典入门）

```js
// f(n) = f(n-1) + f(n-2)
function climbStairs(n) {
  if (n <= 2) return n
  let dp1 = 1, dp2 = 2
  for (let i = 3; i <= n; i++) {
    [dp1, dp2] = [dp2, dp1 + dp2]
  }
  return dp2
}

// 完全背包版本（可爬 1 或 2 步）
// 空间优化：从 O(n) → O(1)
```

### 打家劫舍

```js
// 不能偷连续两间
function rob(nums) {
  if (nums.length === 1) return nums[0]
  const n = nums.length
  const dp = new Array(n)
  dp[0] = nums[0]
  dp[1] = Math.max(nums[0], nums[1])

  for (let i = 2; i < n; i++) {
    dp[i] = Math.max(dp[i - 1], dp[i - 2] + nums[i])
  }
  return dp[n - 1]
}

// 空间优化
function robOpt(nums) {
  let prev2 = 0, prev1 = 0
  for (const num of nums) {
    const curr = Math.max(prev1, prev2 + num)
    prev2 = prev1
    prev1 = curr
  }
  return prev1
}
```

### 股票买卖

```js
// 买卖股票最佳时机（含冷冻期）
function maxProfit(prices) {
  if (prices.length < 2) return 0
  const n = prices.length
  const dp = Array.from({ length: n }, () => [0, 0, 0])
  // dp[i][0]: 持有股票
  // dp[i][1]: 不持有且不在冷冻期（今天卖了）
  // dp[i][2]: 不持有但在冷冻期

  dp[0][0] = -prices[0]  // 买入

  for (let i = 1; i < n; i++) {
    dp[i][0] = Math.max(dp[i - 1][0], dp[i - 1][2] - prices[i])
    dp[i][1] = dp[i - 1][0] + prices[i]
    dp[i][2] = Math.max(dp[i - 1][1], dp[i - 1][2])
  }

  return Math.max(dp[n - 1][1], dp[n - 1][2])
}
```

## 三、路径问题

### 不同路径

```js
// 只能向下或向右，有多少种路径
function uniquePaths(m, n) {
  // dp[i][j] = dp[i-1][j] + dp[i][j-1]
  const dp = Array.from({ length: m }, () => new Array(n).fill(1))

  for (let i = 1; i < m; i++) {
    for (let j = 1; j < n; j++) {
      dp[i][j] = dp[i - 1][j] + dp[i][j - 1]
    }
  }
  return dp[m - 1][n - 1]

  // 空间优化：一维数组
  // const dp = new Array(n).fill(1)
  // for (let i = 1; i < m; i++)
  //   for (let j = 1; j < n; j++)
  //     dp[j] += dp[j - 1]
}

// 含障碍物
function uniquePathsWithObstacles(obstacleGrid) {
  const m = obstacleGrid.length, n = obstacleGrid[0].length
  const dp = new Array(n).fill(0)
  dp[0] = 1

  for (let i = 0; i < m; i++) {
    for (let j = 0; j < n; j++) {
      if (obstacleGrid[i][j] === 1) dp[j] = 0
      else if (j > 0) dp[j] += dp[j - 1]
    }
  }
  return dp[n - 1]
}
```

### 最小路径和

```js
function minPathSum(grid) {
  const m = grid.length, n = grid[0].length
  const dp = Array.from({ length: m }, () => new Array(n))

  for (let i = 0; i < m; i++) {
    for (let j = 0; j < n; j++) {
      if (i === 0 && j === 0) dp[i][j] = grid[0][0]
      else if (i === 0) dp[i][j] = dp[i][j - 1] + grid[i][j]
      else if (j === 0) dp[i][j] = dp[i - 1][j] + grid[i][j]
      else dp[i][j] = Math.min(dp[i - 1][j], dp[i][j - 1]) + grid[i][j]
    }
  }
  return dp[m - 1][n - 1]

  // 空间优化：原地修改 grid
}
```

## 四、子序列问题

### 最长递增子序列（LIS）

```js
// O(n²) 暴力
function lengthOfLIS(nums) {
  const n = nums.length
  const dp = new Array(n).fill(1)
  let maxLen = 1

  for (let i = 1; i < n; i++) {
    for (let j = 0; j < i; j++) {
      if (nums[j] < nums[i]) {
        dp[i] = Math.max(dp[i], dp[j] + 1)
      }
    }
    maxLen = Math.max(maxLen, dp[i])
  }
  return maxLen
}

// O(n log n) 二分优化（维护一个有序数组）
function lengthOfLISBinary(nums) {
  const tails = []
  for (const num of nums) {
    let left = 0, right = tails.length
    while (left < right) {
      const mid = Math.floor((left + right) / 2)
      if (tails[mid] < num) left = mid + 1
      else right = mid
    }
    tails[left] = num
    if (left === tails.length) tails.push(num)
  }
  return tails.length
}
```

### 最长公共子序列（LCS）

```js
function longestCommonSubsequence(text1, text2) {
  const m = text1.length, n = text2.length
  const dp = Array.from({ length: m + 1 }, () => new Array(n + 1).fill(0))

  for (let i = 1; i <= m; i++) {
    for (let j = 1; j <= n; j++) {
      if (text1[i - 1] === text2[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1] + 1
      } else {
        dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1])
      }
    }
  }
  return dp[m][n]
}

// 空间优化：O(n)
```

### 编辑距离

```js
// word1 → word2 最少需要多少次操作
function minDistance(word1, word2) {
  const m = word1.length, n = word2.length
  const dp = Array.from({ length: m + 1 }, (_, i) => [i, ...new Array(n).fill(0)])

  for (let j = 0; j <= n; j++) dp[0][j] = j

  for (let i = 1; i <= m; i++) {
    for (let j = 1; j <= n; j++) {
      if (word1[i - 1] === word2[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1]
      } else {
        dp[i][j] = Math.min(dp[i - 1][j], dp[i][j - 1], dp[i - 1][j - 1]) + 1
        //           删除        插入          替换
      }
    }
  }
  return dp[m][n]
}
```

## 五、背包问题

### 0-1 背包

```js
// 价值最大化：n 个物品，容量 c，每个物品 i 有 weight[i] 和 value[i]
function knapSack01(weights, values, capacity) {
  const n = weights.length
  const dp = Array.from({ length: n + 1 }, () => new Array(capacity + 1).fill(0))

  for (let i = 1; i <= n; i++) {
    for (let w = 0; w <= capacity; w++) {
      if (weights[i - 1] <= w) {
        dp[i][w] = Math.max(
          dp[i - 1][w],
          dp[i - 1][w - weights[i - 1]] + values[i - 1]
        )
      } else {
        dp[i][w] = dp[i - 1][w]
      }
    }
  }
  return dp[n][capacity]

  // 空间优化：一维数组（从后往前遍历避免覆盖）

  //     dp[w] = Math.max(dp[w], dp[w - weights[i]] + values[i])
}
```

### 完全背包

```js
// 每种物品可取无限次
function completeKnapsack(weights, values, capacity) {
  const dp = new Array(capacity + 1).fill(0)

  for (let i = 0; i < weights.length; i++) {
    for (let w = weights[i]; w <= capacity; w++) {
      dp[w] = Math.max(dp[w], dp[w - weights[i]] + values[i])
      // 注意：从前往后遍历（可重复取）
    }
  }
  return dp[capacity]
}

// 零钱兑换 II（组合数）
function change(amount, coins) {
  const dp = new Array(amount + 1).fill(0)
  dp[0] = 1
  for (const coin of coins) {
    for (let i = coin; i <= amount; i++) {
      dp[i] += dp[i - coin]
    }
  }
  return dp[amount]
}
```

## 六、DP 思考步骤

```
1. 明确 dp 数组含义
   → dp[i] 或 dp[i][j] 代表什么？

2. 找出状态转移方程
   → 当前状态怎么由之前状态推导？

3. 确定初始化
   → dp[0]、dp[0][*]、dp[*][0] 如何设置？

4. 确定遍历顺序
   → 从前往后？从后往前？二维的话先行后列？

5. 空间优化（可选）
   → 能否用滚动数组？
   → 0-1 背包：从后往前；完全背包：从前往后
```
