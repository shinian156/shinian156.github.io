## 授权登录概述

### 各平台支持情况

| 登录方式 | 微信小程序 | App（Android）| App（iOS）| H5 |
|---------|----------|--------------|----------|----|
| 微信登录 | ✅ | ✅ | ✅ | ✅ |
| Apple 登录 | ❌ | ❌ | ✅（必须） | ❌ |
| QQ 登录 | ❌ | ✅ | ✅ | ✅ |
| 手机号登录 | ✅（小程序授权） | ✅ | ✅ | ✅ |

> **iOS 强制要求**：App 内提供第三方登录，则必须同时提供 Sign in with Apple。

### 登录流程

```
用户点击第三方登录
       │
       ▼
调用 uni.login 获取 code/authCode
       │
       ▼
将 code 发送到自己的后端
       │
       ▼
后端用 code 换取 openid/access_token
       │
       ▼
查找或创建用户，返回 JWT token
       │
       ▼
前端保存 token，完成登录
```

### 1. 微信小程序登录

```javascript
// utils/auth/wechat.js

/**
 * 微信小程序登录
 * 使用 wx.login 获取 code，换取 openid
 */
export async function wxMiniLogin() {
  return new Promise((resolve, reject) => {
    // #ifdef MP-WEIXIN
    uni.login({
      provider: 'weixin',
      success: async (loginRes) => {
        try {
          const { code } = loginRes

          // 将 code 发送到后端，换取 openid 和 session_key
          const res = await uni.$http.post('/api/auth/wx-mini-login', { code })

          resolve(res.data)
        } catch (err) {
          reject(err)
        }
      },
      fail: reject
    })
    // #endif
  })
}

/**
 * 微信小程序 - 获取手机号（需要用户点击专用按钮）
 */
export async function getWxPhoneNumber(code) {
  // code 来自 <button open-type="getPhoneNumber" @getphonenumber="onGetPhone">
  const res = await uni.$http.post('/api/auth/wx-phone', { code })
  return res.data.phoneNumber
}
```

```javascript
// App 端微信登录（需配置微信 SDK）
export async function wxAppLogin() {
  return new Promise((resolve, reject) => {
    // #ifdef APP-PLUS
    uni.login({
      provider: 'weixin',
      onlyAuthorize: true, // 只获取 authCode，不弹用户信息授权
      success: async (res) => {
        try {
          // App 端拿到的是 authCode
          const { authResult } = res
          const result = await uni.$http.post('/api/auth/wx-app-login', {
            code: authResult.code
          })
          resolve(result.data)
        } catch (err) {
          reject(err)
        }
      },
      fail: reject
    })
    // #endif
  })
}

/**
 * 获取微信用户信息（需用户同意）
 */
export function getWxUserInfo() {
  return new Promise((resolve, reject) => {
    uni.getUserInfo({
      provider: 'weixin',
      success: (res) => {
        resolve({
          nickName: res.userInfo.nickName,
          avatarUrl: res.userInfo.avatarUrl,
          gender: res.userInfo.gender
        })
      },
      fail: reject
    })
  })
}
```

## Apple 授权登录

> iOS 13+ 要求：**只要 App 包含第三方或社交登录，就必须提供 Sign in with Apple**

### 1. 实现 Apple 登录

```javascript
// utils/auth/apple.js

/**
 * Apple 授权登录
 * 仅 iOS 13+ App 端支持
 */
export async function appleLogin() {
  return new Promise((resolve, reject) => {
    // #ifdef APP-PLUS
    const platform = plus.os.name.toLowerCase()
    
    if (platform !== 'ios') {
      reject(new Error('Apple 登录仅支持 iOS'))
      return
    }

    uni.login({
      provider: 'apple',
      success: async (res) => {
        try {
          const {
            user,           // 用户唯一标识（首次登录才返回）
            email,          // 邮箱（首次登录且用户同意才返回）
            fullName,       // 姓名（首次登录且用户同意才返回）
            identityToken,  // JWT token，用于后端验证
            authorizationCode
          } = res.authResult

          const result = await uni.$http.post('/api/auth/apple-login', {
            identityToken,
            authorizationCode,
            user,
            email,
            fullName
          })

          resolve(result.data)
        } catch (err) {
          reject(err)
        }
      },
      fail: (err) => {
        if (err.errMsg.includes('cancel')) {
          reject(new Error('用户取消了 Apple 登录'))
        } else {
          reject(err)
        }
      }
    })
    // #endif
  })
}
```

```json
  "app-plus": {
    "distribute": {
      "sdkConfigs": {
        "oauth": {

        }
      }
    }
  }
}
```

## 统一登录封装

### 登录管理器

```javascript
// utils/auth/index.js
import { wxMiniLogin, wxAppLogin } from './wechat'
import { appleLogin } from './apple'
import { qqLogin } from './qq'

const platform = uni.getSystemInfoSync()

/**
 * 统一登录入口
 * @param {string} provider - 'weixin' | 'apple' | 'qq' | 'phone'
 */
export async function login(provider) {
  uni.showLoading({ title: '登录中...' })
  
  try {
    let data

    switch (provider) {
      case 'weixin':
        if (platform.uniPlatform === 'mp-weixin') {
          data = await wxMiniLogin()
        } else {
          data = await wxAppLogin()
        }
        break

      case 'apple':
        data = await appleLogin()
        break

      case 'qq':
        data = await qqLogin()
        break

      default:
        throw new Error(`不支持的登录方式: ${provider}`)
    }

    // 保存登录状态
    await saveLoginState(data)
    
    uni.hideLoading()
    uni.showToast({ title: '登录成功', icon: 'success' })
    
    return data

  } catch (err) {
    uni.hideLoading()
    
    if (!err.message?.includes('取消')) {
      uni.showToast({ title: err.message || '登录失败', icon: 'none' })
    }
    
    throw err
  }
}

/**
 * 保存登录状态
 */
async function saveLoginState(data) {
  const { token, userInfo, expireTime } = data
  
  uni.setStorageSync('token', token)
  uni.setStorageSync('userInfo', userInfo)
  uni.setStorageSync('tokenExpire', expireTime)
}

/**
 * 检查登录状态
 */
export function checkLogin() {
  const token = uni.getStorageSync('token')
  const expire = uni.getStorageSync('tokenExpire')
  
  if (!token) return false
  
  // 检查 token 是否过期
  if (expire && Date.now() > expire) {
    clearLoginState()
    return false
  }
  
  return true
}

/**
 * 退出登录
 */
export async function logout() {
  try {
    await uni.$http.post('/api/auth/logout')
  } catch {}
  
  clearLoginState()
  
  uni.reLaunch({ url: '/pages/login/index' })
}

function clearLoginState() {
  uni.removeStorageSync('token')
  uni.removeStorageSync('userInfo')
  uni.removeStorageSync('tokenExpire')
}
```

### 登录页面组件

```vue
<!-- pages/login/index.vue -->
<template>
  <view class="login-page">
    <image class="logo" src="/static/logo.png" mode="aspectFit" />
    <text class="title">欢迎登录</text>

      <view class="code-row">
        <input v-model="code" placeholder="验证码" type="number" maxlength="6" />
        <button class="code-btn" :disabled="counting" @click="sendCode">
          {{ counting ? `${countdown}s` : '获取验证码' }}
        </button>
      </view>
      <button class="login-btn primary" @click="handlePhoneLogin">登录</button>
    </view>

    <view class="divider"><text>其他登录方式</text></view>

    <!-- 第三方登录按钮组 -->
    <view class="third-party-login">
      <!-- 微信登录（微信小程序/App） -->
      <view
        v-if="showWechat"
        class="login-icon-btn"
        @click="handleThirdLogin('weixin')"
      >
        <image src="/static/icons/wechat.png" />
        <text>微信</text>
      </view>

      <!-- QQ 登录（App） -->
      <view
        v-if="showQQ"
        class="login-icon-btn"
        @click="handleThirdLogin('qq')"
      >
        <image src="/static/icons/qq.png" />
        <text>QQ</text>
      </view>

      <!-- Apple 登录（iOS 必须提供） -->
      <view
        v-if="showApple"
        class="login-icon-btn apple"
        @click="handleThirdLogin('apple')"
      >
        <image src="/static/icons/apple.png" />
        <text>Apple</text>
      </view>
    </view>

    <text class="agreement">
      登录即表示同意
      <text class="link" @click="toAgreement">《用户协议》</text>
      和
      <text class="link" @click="toPrivacy">《隐私政策》</text>
    </text>
  </view>
</template>

<script setup>
import { ref, computed } from 'vue'
import { login } from '@/utils/auth'

const phone = ref('')
const code = ref('')
const counting = ref(false)
const countdown = ref(60)

const sysInfo = uni.getSystemInfoSync()

// 按平台显示登录方式
const showWechat = computed(() =>
  ['mp-weixin', 'app'].includes(sysInfo.uniPlatform)
)
const showQQ = computed(() =>
  sysInfo.uniPlatform === 'app'
)
const showApple = computed(() =>
  sysInfo.uniPlatform === 'app' && sysInfo.platform === 'ios'
)

async function sendCode() {
  if (!/^1[3-9]\d{9}$/.test(phone.value)) {
    return uni.showToast({ title: '请输入正确手机号', icon: 'none' })
  }
  await uni.$http.post('/api/auth/sms-code', { phone: phone.value })
  startCountdown()
}

function startCountdown() {
  counting.value = true
  countdown.value = 60
  const timer = setInterval(() => {
    if (--countdown.value <= 0) {
      clearInterval(timer)
      counting.value = false
    }
  }, 1000)
}

async function handlePhoneLogin() {
  if (!phone.value || !code.value) {
    return uni.showToast({ title: '请填写完整信息', icon: 'none' })
  }
  try {
    await uni.$http.post('/api/auth/phone-login', {
      phone: phone.value, code: code.value
    })
    uni.navigateBack()
  } catch (err) {
    uni.showToast({ title: err.message || '登录失败', icon: 'none' })
  }
}

async function handleThirdLogin(provider) {
  try {
    await login(provider)
    uni.navigateBack()
  } catch {}
}
</script>
```

## 服务端处理

### Node.js 后端（Koa2）

```javascript
// routes/auth.js
const Router = require('koa-router')
const axios = require('axios')
const jwt = require('jsonwebtoken')
const { verifyAppleToken } = require('../utils/apple-auth')

const router = new Router({ prefix: '/api/auth' })

// 微信小程序登录
router.post('/wx-mini-login', async (ctx) => {
  const { code } = ctx.request.body

  // 用 code 换取 openid 和 session_key
  const wxRes = await axios.get('https://api.weixin.qq.com/sns/jscode2session', {
    params: {
      appid: process.env.WX_MINI_APPID,
      secret: process.env.WX_MINI_SECRET,
      js_code: code,
      grant_type: 'authorization_code'
    }
  })

  const { openid, session_key } = wxRes.data

  if (!openid) {
    ctx.throw(400, '微信授权失败')
  }

  // 查找或创建用户
  let user = await User.findOne({ wxOpenId: openid })
  if (!user) {
    user = await User.create({
      wxOpenId: openid,
      nickname: '微信用户',
      createTime: new Date()
    })
  }

  // 生成 JWT
  const token = jwt.sign(
    { userId: user.id, openid },
    process.env.JWT_SECRET,
    { expiresIn: '7d' }
  )

  ctx.body = {
    code: 0,
    data: {
      token,
      userInfo: { id: user.id, nickname: user.nickname, avatar: user.avatar },
      expireTime: Date.now() + 7 * 24 * 3600 * 1000
    }
  }
})

// Apple 登录验证
router.post('/apple-login', async (ctx) => {
  const { identityToken, user, email, fullName } = ctx.request.body

  // 验证 Apple JWT token
  const appleUser = await verifyAppleToken(identityToken)

  const appleUserId = appleUser.sub // Apple 用户唯一标识

  let dbUser = await User.findOne({ appleUserId })
  if (!dbUser) {
    dbUser = await User.create({
      appleUserId,
      email: email || `${appleUserId}@appleid.com`,
      nickname: fullName ? `${fullName.givenName} ${fullName.familyName}` : 'Apple 用户'
    })
  }

  const token = jwt.sign({ userId: dbUser.id }, process.env.JWT_SECRET, {
    expiresIn: '7d'
  })

  ctx.body = {
    code: 0,
    data: { token, userInfo: dbUser, expireTime: Date.now() + 7 * 24 * 3600 * 1000 }
  }
})

module.exports = router
```

## 安全建议

### 1. token 安全存储

```javascript
// 使用 uni.setStorageSync 存储（加密）
import CryptoJS from 'crypto-js'

const SECRET = 'your-secret-key'

export function secureSet(key, value) {
  const encrypted = CryptoJS.AES.encrypt(
    JSON.stringify(value),
    SECRET
  ).toString()
  uni.setStorageSync(key, encrypted)
}

export function secureGet(key) {
  const encrypted = uni.getStorageSync(key)
  if (!encrypted) return null
  
  try {
    const bytes = CryptoJS.AES.decrypt(encrypted, SECRET)
    return JSON.parse(bytes.toString(CryptoJS.enc.Utf8))
  } catch {
    return null
  }
}
```

### 2. 防止 code 重放

```javascript
// 服务端：code 只能使用一次，使用后立即标记
const usedCodes = new Set()

router.post('/wx-mini-login', async (ctx) => {
  const { code } = ctx.request.body
  
  if (usedCodes.has(code)) {
    ctx.throw(400, 'code 已使用')
  }
  
  usedCodes.add(code)
  setTimeout(() => usedCodes.delete(code), 5 * 60 * 1000)
  
  // ... 正常处理
})
```

### 3. 验证 token 有效性

```javascript
// 请求拦截器：token 自动续期
uni.$http.interceptors.request.use((config) => {
  const token = uni.getStorageSync('token')
  const expire = uni.getStorageSync('tokenExpire')
  
  if (token) {
    config.header.Authorization = `Bearer ${token}`
  }
  
  // token 即将过期，触发静默续期
  if (expire && expire - Date.now() < 24 * 3600 * 1000) {
    refreshToken()
  }
  
  return config
})

async function refreshToken() {
  try {
    const res = await uni.$http.post('/api/auth/refresh')
    uni.setStorageSync('token', res.data.token)
    uni.setStorageSync('tokenExpire', res.data.expireTime)
  } catch {
    // 续期失败，重新登录
    logout()
  }
}
```
