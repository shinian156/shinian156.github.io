```js
// 非递归前序
function preorderIter(root) {
  const stack = [], result = []
  let curr = root
  while (curr || stack.length) {
    while (curr) {
      result.push(curr.val)
      stack.push(curr)
      curr = curr.left
    }
    curr = stack.pop().right
  }
  return result
}

// 非递归中序
function inorderIter(root) {
  const stack = [], result = []
  let curr = root
  while (curr || stack.length) {
    while (curr) {
      stack.push(curr)
      curr = curr.left
    }
    curr = stack.pop()
    result.push(curr.val)
    curr = curr.right
  }
  return result
}

// 层序遍历（BFS）
function levelOrder(root) {
  if (!root) return []
  const queue = [root], result = []
  while (queue.length) {
    const level = [], size = queue.length
    for (let i = 0; i < size; i++) {
      const node = queue.shift()
      level.push(node.val)
      if (node.left) queue.push(node.left)
      if (node.right) queue.push(node.right)
    }
    result.push(level)
  }
  return result
}
```

## 二、二叉树经典题目

### 求深度

```js
// 最大深度（后序遍历）
function maxDepth(root) {
  if (!root) return 0
  return Math.max(maxDepth(root.left), maxDepth(root.right)) + 1
}

// 最小深度（注意叶子节点）
function minDepth(root) {
  if (!root) return 0
  if (!root.left) return minDepth(root.right) + 1
  if (!root.right) return minDepth(root.left) + 1
  return Math.min(minDepth(root.left), minDepth(root.right)) + 1
}

// 判断平衡二叉树
function isBalanced(root) {
  function dfs(node) {
    if (!node) return 0
    const left = dfs(node.left)
    if (left === -1) return -1
    const right = dfs(node.right)
    if (right === -1) return -1
    if (Math.abs(left - right) > 1) return -1
    return Math.max(left, right) + 1
  }
  return dfs(root) !== -1
}
```

### 路径相关

```js
// 二叉树直径（任意两节点最长路径）
function diameterOfBinaryTree(root) {
  let diameter = 0
  function depth(node) {
    if (!node) return 0
    const left = depth(node.left)
    const right = depth(node.right)
    diameter = Math.max(diameter, left + right)
    return Math.max(left, right) + 1
  }
  depth(root)
  return diameter
}

// 路径总和（是否存在根到叶子的和为 targetSum）
function hasPathSum(root, target) {
  if (!root) return false
  if (!root.left && !root.right) return root.val === target
  return hasPathSum(root.left, target - root.val)
       || hasPathSum(root.right, target - root.val)
}

// 所有路径
function binaryTreePaths(root) {
  const result = []
  function dfs(node, path) {
    if (!node) return
    path += node.val
    if (!node.left && !node.right) result.push(path)
    else {
      path += '->'
      dfs(node.left, path)
      dfs(node.right, path)
    }
  }
  dfs(root, '')
  return result
}
```

## 三、堆（完全二叉树）

```js
class MinHeap {
  constructor() { this.heap = [] }

  getParent(i) { return Math.floor((i - 1) / 2) }
  getLeft(i) { return 2 * i + 1 }
  getRight(i) { return 2 * i + 2 }

  swap(i, j) { [this.heap[i], this.heap[j]] = [this.heap[j], this.heap[i]] }

  insert(val) {
    this.heap.push(val)
    this.bubbleUp(this.heap.length - 1)
  }

  bubbleUp(i) {
    while (i > 0 && this.heap[i] < this.heap[this.getParent(i)]) {
      this.swap(i, this.getParent(i))
      i = this.getParent(i)
    }
  }

  extractMin() {
    if (!this.heap.length) return null
    const min = this.heap[0]
    this.heap[0] = this.heap.pop()
    this.bubbleDown(0)
    return min
  }

  bubbleDown(i) {
    while (this.getLeft(i) < this.heap.length) {
      let smallest = this.getLeft(i)
      if (this.getRight(i) < this.heap.length
          && this.heap[this.getRight(i)] < this.heap[smallest]) {
        smallest = this.getRight(i)
      }
      if (this.heap[i] <= this.heap[smallest]) break
      this.swap(i, smallest)
      i = smallest
    }
  }
}
```

```js
class Graph {

    if (!this.adj[v1]) this.adj[v1] = []
    if (!this.adj[v2]) this.adj[v2] = []
```

## 五、图的遍历

### BFS

```js
function bfs(graph, start) {
  const visited = new Set()
  const queue = [start]
  const result = []

  while (queue.length) {
    const v = queue.shift()
    if (visited.has(v)) continue
    visited.add(v)
    result.push(v)
    for (const neighbor of graph[v] || []) {
      if (!visited.has(neighbor)) queue.push(neighbor)
    }
  }
  return result
}

// 最短路径（无权图）
function shortestPath(graph, start, end) {
  const visited = new Set()
  const queue = [[start, [start]]]
  visited.add(start)

  while (queue.length) {
    const [v, path] = queue.shift()
    if (v === end) return path
    for (const neighbor of graph[v] || []) {
      if (!visited.has(neighbor)) {
        visited.add(neighbor)
        queue.push([neighbor, [...path, neighbor]])
      }
    }
  }
  return []
}
```

```js
function dfsIterative(graph, start) {
  const visited = new Set()
  const stack = [start]
  const result = []

  while (stack.length) {
    const v = stack.pop()
    if (visited.has(v)) continue
    visited.add(v)
    result.push(v)
    for (const neighbor of (graph[v] || []).reverse()) {
      if (!visited.has(neighbor)) stack.push(neighbor)
    }
  }
  return result
}

// 岛屿数量
function numIslands(grid) {
  let count = 0
  const m = grid.length, n = grid[0].length

  function dfs(i, j) {
    if (i < 0 || i >= m || j < 0 || j >= n || grid[i][j] === '0') return
    grid[i][j] = '0'  // 标记已访问
    dfs(i + 1, j); dfs(i - 1, j)
    dfs(i, j + 1); dfs(i, j - 1)
  }

  for (let i = 0; i < m; i++) {
    for (let j = 0; j < n; j++) {
      if (grid[i][j] === '1') {
        count++
        dfs(i, j)
      }
    }
  }
  return count
}
```

## 六、拓扑排序

```js
// Kahn 算法（BFS）
function topologicalSortKahn(numCourses, prerequisites) {
  const graph = {}, indegree = Array(numCourses).fill(0)

  for (const [course, prereq] of prerequisites) {
    if (!graph[prereq]) graph[prereq] = []
    graph[prereq].push(course)
    indegree[course]++
  }

  const queue = []
  for (let i = 0; i < numCourses; i++) {
    if (indegree[i] === 0) queue.push(i)
  }

  const result = []
  while (queue.length) {
    const v = queue.shift()
    result.push(v)
    for (const neighbor of graph[v] || []) {
      indegree[neighbor]--
      if (indegree[neighbor] === 0) queue.push(neighbor)
    }
  }

  return result.length === numCourses ? result : []
}

// DFS 版本
function topologicalSortDFS(numCourses, prerequisites) {
  const graph = {}, visited = {}, onStack = {}, result = []

  for (const [course, prereq] of prerequisites) {
    if (!graph[prereq]) graph[prereq] = []
    graph[prereq].push(course)
  }

  function dfs(v) {
    visited[v] = true
    onStack[v] = true
    for (const neighbor of graph[v] || []) {
      if (!visited[neighbor] && dfs(neighbor)) return true
      if (onStack[neighbor]) return true  // 有环
    }
    onStack[v] = false
    result.push(v)
    return false
  }

  for (let i = 0; i < numCourses; i++) {
    if (!visited[i] && dfs(i)) return []
  }
  return result.reverse()
}
```

## 七、并查集（Union-Find）

```js
class UnionFind {
  constructor(n) {
    this.parent = Array.from({ length: n }, (_, i) => i)
    this.rank = Array(n).fill(1)
    this.count = n
  }

  find(x) {
    if (this.parent[x] !== x) this.parent[x] = this.find(this.parent[x])
    return this.parent[x]
  }

  union(x, y) {
    const px = this.find(x), py = this.find(y)
    if (px === py) return false
    if (this.rank[px] < this.rank[py]) [px, py] = [py, px]
    this.parent[py] = px
    if (this.rank[px] === this.rank[py]) this.rank[px]++
    this.count--
    return true
  }

  connected(x, y) { return this.find(x) === this.find(y) }
}

// 应用：朋友圈数量
function findCircleNum(M) {
  const n = M.length
  const uf = new UnionFind(n)
  for (let i = 0; i < n; i++)
    for (let j = i + 1; j < n; j++)
      if (M[i][j] === 1) uf.union(i, j)
  return uf.count
}
```
