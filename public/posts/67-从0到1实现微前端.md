## 项目规划

本文将使用 **qiankun** 从零搭建一个完整的微前端系统，包含：

```
micro-frontend-demo/
├── main-app/           # 主应用（Vue 3 + Vite）
├── sub-app-vue/        # 子应用A（Vue 3）
├── sub-app-react/      # 子应用B（React 18）
└── shared/             # 共享工具库
```

### 最终效果

```
访问 http://localhost:3000/         → 主应用首页
访问 http://localhost:3000/vue-app  → 加载 Vue 子应用
访问 http://localhost:3000/react-app → 加载 React 子应用
```

### 4. 初始化 qiankun

```typescript
// src/micro/lifecycle.ts
import { registerMicroApps, start, addGlobalUncaughtErrorHandler } from 'qiankun'
import { microApps } from './apps'
import { globalStore } from '../store/global'

export function setupMicroApps() {
  registerMicroApps(
    microApps.map(app => ({
      ...app,
      props: {
        ...app.props,
        // 注入全局状态和通信方法
        globalStore,
        onMicroAppEvent: (event: string, data: any) => {
          globalStore.handleSubAppEvent(event, data)
        }
      }
    })),
    {
      // 全局生命周期钩子
      beforeLoad: [
        async (app) => {
          console.log(`[主应用] ${app.name} 开始加载`)
          globalStore.setLoading(true)
        }
      ],
      beforeMount: [
        async (app) => {
          console.log(`[主应用] ${app.name} 开始挂载`)
        }
      ],
      afterMount: [
        async (app) => {
          console.log(`[主应用] ${app.name} 挂载完成`)
          globalStore.setLoading(false)
        }
      ],
      afterUnmount: [
        async (app) => {
          console.log(`[主应用] ${app.name} 已卸载`)
        }
      ]
    }
  )

  // 全局异常处理
  addGlobalUncaughtErrorHandler((event) => {
    const { msg, error } = event as any
    if (msg && msg.includes('died in status LOADING_SOURCE_CODE')) {
      console.error('子应用加载失败，请检查应用是否正常运行:', error)
    }
  })

  start({
    sandbox: {
      experimentalStyleIsolation: true // 开启样式隔离
    },
    prefetch: true // 预加载
  })
}
```

### 5. 全局状态管理

```typescript
// src/store/global.ts
import { reactive } from 'vue'

interface GlobalState {
  user: {
    id: number
    name: string
    token: string
  } | null
  loading: boolean
  theme: 'light' | 'dark'
}

const state = reactive<GlobalState>({
  user: null,
  loading: false,
  theme: 'light'
})

export const globalStore = {
  // 读取状态
  getState: () => ({ ...state }),

  // 用户操作
  setUser(user: GlobalState['user']) {
    state.user = user
  },
  logout() {
    state.user = null
    localStorage.removeItem('token')
  },

  // 加载状态
  setLoading(loading: boolean) {
    state.loading = loading
  },

  // 处理子应用事件
  handleSubAppEvent(event: string, data: any) {
    console.log(`[主应用] 收到子应用事件: ${event}`, data)
    switch (event) {
      case 'USER_LOGIN':
        this.setUser(data)
        break
      case 'USER_LOGOUT':
        this.logout()
        break
      case 'NAVIGATE':
        // 跨应用导航
        window.history.pushState(null, '', data.path)
        break
    }
  },

  // 主题切换
  toggleTheme() {
    state.theme = state.theme === 'light' ? 'dark' : 'light'
    document.body.setAttribute('data-theme', state.theme)
  }
}
```

#### 3. 子应用 main.ts 改造

```typescript
// sub-app-vue/src/main.ts
import { createApp, App as VueApp } from 'vue'
import { createRouter, createWebHistory } from 'vue-router'
import App from './App.vue'
import routes from './router/routes'
import {
  renderWithQiankun,
  qiankunWindow
} from 'vite-plugin-qiankun/dist/helper'

let app: VueApp | null = null
let router: any = null

function render(props: any = {}) {
  const { container, routerBase = '/' } = props

  router = createRouter({
    history: createWebHistory(
      qiankunWindow.__POWERED_BY_QIANKUN__ ? routerBase : '/'
    ),
    routes
  })

  app = createApp(App)

  // 注入主应用传入的 props
  app.provide('mainAppProps', props)
  app.provide('globalStore', props.globalStore)

  app.use(router)

  const mountTarget = container
    ? container.querySelector('#app')
    : document.querySelector('#app')

  app.mount(mountTarget!)
}

// 独立运行时
if (!qiankunWindow.__POWERED_BY_QIANKUN__) {
  render()
}

// 在 qiankun 环境中运行
renderWithQiankun({
  mount(props) {
    render(props)
  },
  bootstrap() {
    console.log('[sub-app-vue] bootstrap')
  },
  unmount() {
    app?.unmount()
    app = null
    router = null
  },
  update(props) {
    console.log('[sub-app-vue] update props', props)
  }
})
```

// sub-app-vue/src/composables/useGlobalStore.ts
import { inject, computed } from 'vue'
import { qiankunWindow } from 'vite-plugin-qiankun/dist/helper'

export function useGlobalStore() {
  // 获取主应用注入的 globalStore
  const globalStore = inject<any>('globalStore')

  // 如果不在 qiankun 环境，使用本地 mock
  if (!qiankunWindow.__POWERED_BY_QIANKUN__) {
    return {
      user: { id: 1, name: '测试用户', token: 'mock-token' },
      sendEvent: (event: string, data: any) => console.log(event, data)
    }
  }

  const user = computed(() => globalStore?.getState()?.user)

  const sendEvent = (event: string, data: any) => {
    const onMicroAppEvent = inject<Function>('onMicroAppEvent' as any)
    onMicroAppEvent?.(event, data)
  }

  return { user, sendEvent }
}

### 方案二：自定义事件总线

```typescript
// shared/event-bus.ts - 跨应用事件总线
class EventBus {
  private events: Map<string, Set<Function>> = new Map()

  on(event: string, callback: Function) {
    if (!this.events.has(event)) {
      this.events.set(event, new Set())
    }
    this.events.get(event)!.add(callback)
    
    // 返回取消订阅函数
    return () => this.off(event, callback)
  }

  off(event: string, callback: Function) {
    this.events.get(event)?.delete(callback)
  }

  emit(event: string, data?: any) {
    this.events.get(event)?.forEach(cb => cb(data))
  }

  once(event: string, callback: Function) {
    const unsubscribe = this.on(event, (...args: any[]) => {
      callback(...args)
      unsubscribe()
    })
  }
}

// 挂载到 window，跨应用共享
declare global {
  interface Window {
    __MICRO_FRONTEND_BUS__: EventBus
  }
}

window.__MICRO_FRONTEND_BUS__ = window.__MICRO_FRONTEND_BUS__ || new EventBus()
export const eventBus = window.__MICRO_FRONTEND_BUS__
```

```typescript
// 使用示例
import { eventBus } from 'shared/event-bus'

// 子应用 A 发送事件
eventBus.emit('USER_UPDATED', { id: 1, name: '新名字' })

// 子应用 B 监听事件
const unsubscribe = eventBus.on('USER_UPDATED', (user) => {
  console.log('用户信息更新:', user)
})

// 组件卸载时取消订阅
onUnmounted(unsubscribe)
```

## 部署方案

### Nginx 配置

```nginx
# nginx.conf - 微前端部署配置

server {
    listen 80;
    server_name your-domain.com;

    # 主应用
    location / {
        root /var/www/main-app;
        try_files $uri $uri/ /index.html;
    }

    # Vue 子应用（独立域名路径）
    location /vue-app/ {
        root /var/www;
        try_files $uri $uri/ /vue-app/index.html;
        
        # 允许跨域（供主应用 HTML Entry 访问）
        add_header Access-Control-Allow-Origin *;
        add_header Access-Control-Allow-Methods 'GET, POST, OPTIONS';
        add_header Access-Control-Allow-Headers 'DNT,X-Mx-ReqToken,Keep-Alive,User-Agent,X-Requested-With,If-Modified-Since,Cache-Control,Content-Type';
    }

    # React 子应用
    location /react-app/ {
        root /var/www;
        try_files $uri $uri/ /react-app/index.html;
        add_header Access-Control-Allow-Origin *;
    }
    
    # 静态资源缓存
    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
    }
}
```

### 问题3：沙箱中 setTimeout/setInterval 不清理

```typescript
// 子应用 unmount 时手动清理
let timer: number | null = null

export async function mount() {
  timer = window.setInterval(() => {
    // 定时任务
  }, 1000)
}

export async function unmount() {
  if (timer) {
    clearInterval(timer)
    timer = null
  }
}
```

### 微前端核心原则

> - **独立性优先**：子应用能独立运行，不依赖主应用
> - **最小化通信**：应用间通信越少越好，降低耦合
> - **隔离要彻底**：JS 沙箱 + 样式隔离缺一不可
> - **渐进式接入**：老系统可逐步迁移，不要一次全改
